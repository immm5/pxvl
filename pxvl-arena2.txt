(() => {
  if (window.pvArena2) {
    console.warn("pvArena2 already loaded.");
    return;
  }

  const CFG = {
    startPath: "/api/arena/start",
    upgradeBowAttackPath: "/api/character/upgrade-bow-attack",
    upgradeMiningEfficiencyPath: "/api/character/upgrade-mining-efficiency",
    startTimeoutMs: 15000,
    cooldownExtraWaitMs: 15000,
    preStartAttachTimeoutMs: 5000,
    activeArenaWaitTimeoutMs: 900000,
    wsPath: "/api/arena/ws",
    wsHostPattern: "pixelvalley",
    autoStartOnLoad: true,
    autoDailyEnabled: true,
    restartDelayMinMs: 25000,
    restartDelayMaxMs: 35000,
    maxAutoRestarts: 12,
    initialPosition: { gridX: 10, gridY: 9 },
    attackBurst: 16,
    attackDelayMinMs: 333,
    attackDelayMaxMs: 444,
    betweenBurstMs: 120,
    hpResumeThresholdPct: 0.75,
    hpRefillPollMs: 500,
    hpRefillWaitTimeoutMs: 120000,
    afterKillCooldownMs: 10000,
    chainAttackMaxDistance: 2,
    cooldownThreatDistance: 2,
    approachThreatRadius: 3,
    approachDirectionalRadius: 6,
    approachDirectionalPenalty: 120,
    maxAdditionalThreatForEngage: 0,
    fallbackAdditionalThreatForEngage: 1,
    safeEngageStallCycles: 10,
    crowdRepositionWaitMs: 350,
    positionDelayMs: 60,
    acquireTimeoutMs: 8000
  };

  const sleep = (ms) => new Promise((r) => setTimeout(r, ms));
  const now = () => new Date().toISOString().slice(11, 19);
  const randInt = (min, max) => Math.floor(Math.random() * (max - min + 1)) + min;
  const extractTokenFromUrl = (url) => {
    if (!url || typeof url !== "string") return null;
    try {
      const u = new URL(url, location.origin);
      const t = u.searchParams.get("token");
      return t || null;
    } catch {
      return null;
    }
  };
  const safeJson = (raw) => {
    try {
      return JSON.parse(raw);
    } catch {
      return raw;
    }
  };

  const state = {
    stopped: false,
    running: false,
    activeSocket: null,
    latestArena: null,
    playerPos: null,
    installed: false,
    token: null,
    runId: null,
    lastWsUrl: null,
    lastStart: null,
    battlesLeft: null,
    maxBattlesPerDay: null,
    autoDailyEnabled: CFG.autoDailyEnabled,
    autoRestarting: false,
    autoRestartCount: 0,
    startInFlight: false,
    socketGeneration: 0,
    cooldownRetryTimer: null,
    cooldownRetryAt: null,
    awaitingActiveArena: false,
    lastBattleOutcome: null
  };

  const orig = {
    WebSocket: window.WebSocket,
    send: WebSocket.prototype.send
  };

  const log = (...args) => console.log(`[pvArena2 ${now()}]`, ...args);
  const warn = (...args) => console.warn(`[pvArena2 ${now()}]`, ...args);

  const wsOrigin = () => `${location.protocol === "https:" ? "wss:" : "ws:"}//${location.host}`;
  const isArenaUrl = (url) => {
    if (typeof url !== "string") return false;
    const lower = url.toLowerCase();
    return lower.startsWith("ws") && lower.includes(CFG.wsPath) && lower.includes(CFG.wsHostPattern);
  };

  const parseNumber = (value) => {
    const n = Number(value);
    return Number.isFinite(n) ? n : null;
  };

  const setBattleOutcomeFromMessage = (data) => {
    if (data?.victory === true) {
      state.lastBattleOutcome = {
        type: "victory",
        playerHp: parseNumber(data.playerHp ?? state.latestArena?.playerHp),
        xpGained: parseNumber(data.xpGained),
        cooldownUntil: typeof data?.cooldownUntil === "string" ? data.cooldownUntil : null,
        at: new Date().toISOString()
      };
      return;
    }

    if (data?.defeat === true) {
      state.lastBattleOutcome = {
        type: "defeat",
        playerHp: parseNumber(data.playerHp ?? state.latestArena?.playerHp),
        xpGained: parseNumber(data.xpGained),
        cooldownUntil: typeof data?.cooldownUntil === "string" ? data.cooldownUntil : null,
        at: new Date().toISOString()
      };
      return;
    }

    if (data?.error === "cooldown" && typeof data?.cooldownUntil === "string") {
      if (!state.lastBattleOutcome || state.lastBattleOutcome.type === "cooldown") {
        state.lastBattleOutcome = {
          type: "cooldown",
          playerHp: parseNumber(data.playerHp ?? state.latestArena?.playerHp),
          xpGained: null,
          cooldownUntil: data.cooldownUntil,
          at: new Date().toISOString()
        };
      }
    }
  };

  const formatBattleOutcomeForLog = (outcome) => {
    if (!outcome) return "result=UNKNOWN";
    const parts = [];
    if (outcome.type === "victory") parts.push("result=WIN");
    else if (outcome.type === "defeat") parts.push("result=LOSE");
    else parts.push("result=COOLDOWN");

    if (Number.isFinite(outcome.playerHp)) parts.push(`playerHp=${outcome.playerHp}`);
    if (Number.isFinite(outcome.xpGained)) parts.push(`xp=${outcome.xpGained}`);
    if (outcome.cooldownUntil) parts.push(`cooldownUntil=${outcome.cooldownUntil}`);
    return parts.join(" | ");
  };

  const toPos = (x, y) => {
    const px = parseNumber(x);
    const py = parseNumber(y);
    return px === null || py === null ? null : { x: px, y: py };
  };

  const extractPlayerPosFromState = (data) => {
    const pairs = [
      ["playerTileX", "playerTileY"],
      ["playerGridX", "playerGridY"],
      ["playerX", "playerY"],
      ["gridX", "gridY"],
      ["tileX", "tileY"]
    ];

    for (const [kx, ky] of pairs) {
      const p = toPos(data?.[kx], data?.[ky]);
      if (p) return p;
    }

    const nested = [data?.player, data?.position, data?.character];
    for (const obj of nested) {
      if (!obj || typeof obj !== "object") continue;
      const p1 = toPos(obj.tileX, obj.tileY);
      const p2 = toPos(obj.gridX, obj.gridY);
      const p3 = toPos(obj.x, obj.y);
      if (p1) return p1;
      if (p2) return p2;
      if (p3) return p3;
    }

    return null;
  };

  const updatePlayerPosFromState = (data) => {
    const pos = extractPlayerPosFromState(data);
    if (pos) state.playerPos = pos;
  };

  const parseIncoming = (raw) => {
    const data = safeJson(typeof raw === "string" ? raw : String(raw));
    if (!data || typeof data !== "object") return;
    setBattleOutcomeFromMessage(data);

    if (Array.isArray(data.monsters)) {
      state.latestArena = data;
      updatePlayerPosFromState(data);
      runLoop();
    }
  };

  const attachArenaSocket = (ws, source) => {
    if (!ws || ws.__pvArena2Attached) return;
    ws.__pvArena2Attached = true;
    ws.__pvArena2InitPosSent = false;
    ws.__pvArena2SocketGeneration = ++state.socketGeneration;
    state.activeSocket = ws;
    state.lastBattleOutcome = null;

    ws.addEventListener("message", (ev) => parseIncoming(ev.data));
    ws.addEventListener("close", (ev) => {
      if (state.activeSocket === ws) state.activeSocket = null;
      state.socketGeneration += 1;
      state.running = false;
      log(
        "WSS close:",
        ev?.code ?? "?",
        ev?.reason || "(no reason)",
        "|",
        formatBattleOutcomeForLog(state.lastBattleOutcome)
      );
      if (!ws.__pvArena2SkipRestart) {
        maybeAutoRestartAfterClose(ev);
      }
    });
    ws.addEventListener("error", (ev) => warn("Arena socket error:", ev));

    log(`Arena socket attached from ${source}:`, ws.url);
  };

  const setAutoDaily = (enabled) => {
    state.autoDailyEnabled = Boolean(enabled);
    log("Auto daily:", state.autoDailyEnabled ? "ON" : "OFF");
    return state.autoDailyEnabled;
  };

  const clearCooldownRetry = () => {
    if (state.cooldownRetryTimer) {
      clearTimeout(state.cooldownRetryTimer);
      state.cooldownRetryTimer = null;
    }
    state.cooldownRetryAt = null;
  };

  const scheduleCooldownRetry = (token, cooldownUntil) => {
    const ts = Date.parse(cooldownUntil || "");
    if (!Number.isFinite(ts)) {
      warn("Cooldown terdeteksi tapi cooldownUntil invalid:", cooldownUntil);
      return false;
    }

    const waitMs = Math.max(0, ts - Date.now()) + CFG.cooldownExtraWaitMs;
    clearCooldownRetry();
    state.cooldownRetryAt = new Date(Date.now() + waitMs).toISOString();

    log(
      `Cooldown aktif. Retry start dalam ${Math.ceil(waitMs / 1000)}s ` +
      `(cooldownUntil + ${CFG.cooldownExtraWaitMs / 1000}s) at ${state.cooldownRetryAt}`
    );

    state.cooldownRetryTimer = setTimeout(async () => {
      state.cooldownRetryTimer = null;
      state.cooldownRetryAt = null;

      if (state.stopped) {
        log("Cooldown retry dibatalkan karena state stopped.");
        return;
      }

      try {
        await startArena({ token, autoConnect: true });
      } catch (e) {
        warn("Cooldown retry error:", e?.message || e);
      }
    }, waitMs);

    return true;
  };

  const maybeAutoRestartAfterClose = async (_ev) => {
    if (!state.autoDailyEnabled) return;
    if (state.stopped) return;
    if (state.autoRestarting) return;
    if (state.startInFlight) return;

    const left = Number(state.battlesLeft);
    if (Number.isFinite(left) && left <= 0) {
      log("Daily battles exhausted (battlesLeft=0). Auto stop.");
      return;
    }

    if (state.autoRestartCount >= CFG.maxAutoRestarts) {
      warn(`Auto restart limit reached (${CFG.maxAutoRestarts}). Auto stop.`);
      return;
    }

    state.autoRestarting = true;
    state.autoRestartCount += 1;
    const waitMs = randInt(CFG.restartDelayMinMs, CFG.restartDelayMaxMs);
    log(`Auto restart #${state.autoRestartCount} in ${waitMs}ms...`);
    await sleep(waitMs);

    try {
      if (state.stopped) return;
      const result = await startArena({ autoConnect: true });
      if (!result?.ok) {
        if (result?.cooldownScheduled) {
          log("Start kena cooldown. Retry sudah dijadwalkan:", result.cooldownRetryAt);
        } else {
          warn("Auto restart start() failed:", result);
        }
      }
    } catch (e) {
      warn("Auto restart error:", e?.message || e);
    } finally {
      state.autoRestarting = false;
    }
  };

  const sendInitialPositionOnce = () => {
    const ws = state.activeSocket;
    if (!ws || ws.readyState !== WebSocket.OPEN) return false;
    if (ws.__pvArena2InitPosSent) return true;

    const initPos = {
      type: "position",
      gridX: CFG.initialPosition.gridX,
      gridY: CFG.initialPosition.gridY
    };

    try {
      ws.send(JSON.stringify(initPos));
      ws.__pvArena2InitPosSent = true;
      state.playerPos = { x: initPos.gridX, y: initPos.gridY };
      log("Initial position sent once:", initPos);
      return true;
    } catch (e) {
      warn("Gagal kirim initial position:", e);
      return false;
    }
  };

  const installHooks = () => {
    if (state.installed) return;
    state.installed = true;

    WebSocket.prototype.send = function patchedSend(data) {
      try {
        if (isArenaUrl(this.url)) {
          attachArenaSocket(this, "prototype.send");
        }

        if (typeof data === "string") {
          const parsed = JSON.parse(data);
          if (parsed?.type === "position") {
            const p = toPos(parsed.gridX, parsed.gridY);
            if (p) state.playerPos = p;
          }
        }
      } catch {}
      return orig.send.call(this, data);
    };

    const NativeWS = window.WebSocket;
    function PatchedWebSocket(url, protocols) {
      const ws = protocols === undefined ? new NativeWS(url) : new NativeWS(url, protocols);
      try {
        if (isArenaUrl(ws.url || url)) attachArenaSocket(ws, "constructor");
      } catch {}
      return ws;
    }

    PatchedWebSocket.prototype = NativeWS.prototype;
    Object.setPrototypeOf(PatchedWebSocket, NativeWS);
    window.WebSocket = PatchedWebSocket;
  };

  const buildWsUrl = (token, runId) =>
    `${wsOrigin()}${CFG.wsPath}?token=${encodeURIComponent(token)}&runId=${encodeURIComponent(runId)}`;

  const findArenaUrlFromPerformance = () => {
    try {
      const entries = performance.getEntriesByType("resource") || [];
      for (let i = entries.length - 1; i >= 0; i -= 1) {
        const name = entries[i]?.name;
        if (isArenaUrl(name)) return name;
      }
    } catch {}
    return null;
  };

  const waitForSocketOpen = (ws, timeoutMs) =>
    new Promise((resolve) => {
      if (!ws) return resolve(false);
      if (ws.readyState === WebSocket.OPEN) return resolve(true);
      if (ws.readyState >= WebSocket.CLOSING) return resolve(false);

      let settled = false;
      const done = (ok) => {
        if (settled) return;
        settled = true;
        clearTimeout(timer);
        resolve(ok);
      };
      const onOpen = () => done(true);
      const onClose = () => done(false);
      const onError = () => done(false);
      const timer = setTimeout(() => done(false), timeoutMs);
      ws.addEventListener("open", onOpen, { once: true });
      ws.addEventListener("close", onClose, { once: true });
      ws.addEventListener("error", onError, { once: true });
    });

  const waitForSocketClose = (ws, timeoutMs) =>
    new Promise((resolve) => {
      if (!ws || ws.readyState >= WebSocket.CLOSING) return resolve(true);
      let settled = false;
      const done = (ok) => {
        if (settled) return;
        settled = true;
        clearTimeout(timer);
        resolve(ok);
      };
      const timer = setTimeout(() => done(false), timeoutMs);
      ws.addEventListener("close", () => done(true), { once: true });
    });

  const ensureNoActiveArenaBeforeStart = async () => {
    state.awaitingActiveArena = true;
    try {
      let ws = state.activeSocket;
      if (ws && ws.readyState === WebSocket.OPEN) {
        ws.__pvArena2SkipRestart = true;
        log("Arena aktif terdeteksi. Selesaikan dulu sebelum start baru...");
        runLoop();
        const closed = await waitForSocketClose(ws, CFG.activeArenaWaitTimeoutMs);
        if (!closed) warn("Timeout menunggu arena aktif selesai. Lanjut start baru.");
        return;
      }

      const perfUrl = findArenaUrlFromPerformance();
      if (!perfUrl) return;
      if (state.activeSocket?.url === perfUrl && state.activeSocket?.readyState === WebSocket.OPEN) return;

      log("Mencoba attach arena aktif dari performance WSS sebelum start...");
      ws = new orig.WebSocket(perfUrl);
      ws.__pvArena2SkipRestart = true;
      ws.addEventListener("open", () => {
        attachArenaSocket(ws, "performance.prestart");
        sendInitialPositionOnce();
        runLoop();
      });

      const opened = await waitForSocketOpen(ws, CFG.preStartAttachTimeoutMs);
      if (!opened) {
        try { ws.close(); } catch {}
        return;
      }

      const closed = await waitForSocketClose(ws, CFG.activeArenaWaitTimeoutMs);
      if (!closed) warn("Timeout menunggu arena (performance attach) selesai. Lanjut start baru.");
    } finally {
      state.awaitingActiveArena = false;
    }
  };

  const setToken = (token) => {
    state.token = token;
    return token;
  };

  const isLikelyToken = (value) => {
    if (typeof value !== "string") return false;
    const v = value.trim();
    if (!v || v.length < 20) return false;
    if (/\s/.test(v)) return false;
    return true;
  };

  const maybeTokenFromValue = (value) => {
    if (typeof value !== "string") return null;
    const trimmed = value.trim();
    if (isLikelyToken(trimmed)) return trimmed;
    if (/^bearer\s+/i.test(trimmed)) {
      const t = trimmed.replace(/^bearer\s+/i, "").trim();
      if (isLikelyToken(t)) return t;
    }
    return null;
  };

  const scanStorageForToken = (storage) => {
    if (!storage) return null;
    const keyHints = ["token", "auth", "access", "bearer", "jwt", "session"];
    const objHints = ["token", "accessToken", "authToken", "bearer", "authorization"];
    for (let i = 0; i < storage.length; i += 1) {
      const key = storage.key(i);
      if (!key) continue;
      let value = "";
      try {
        value = storage.getItem(key) ?? "";
      } catch {
        continue;
      }

      const direct = maybeTokenFromValue(value);
      if (direct && keyHints.some((h) => key.toLowerCase().includes(h))) return direct;

      if (value.startsWith("{") || value.startsWith("[")) {
        const parsed = safeJson(value);
        if (parsed && typeof parsed === "object") {
          for (const hint of objHints) {
            const nested = parsed?.[hint];
            const token = maybeTokenFromValue(typeof nested === "string" ? nested : "");
            if (token) return token;
          }
        }
      }
    }
    return null;
  };

  const scanCookieForToken = () => {
    let raw = "";
    try {
      raw = document.cookie || "";
    } catch {
      return null;
    }
    if (!raw) return null;

    const keyHints = ["token", "auth", "access", "bearer", "session"];
    const parts = raw.split(";").map((s) => s.trim()).filter(Boolean);
    for (const part of parts) {
      const idx = part.indexOf("=");
      if (idx < 1) continue;
      const key = decodeURIComponent(part.slice(0, idx)).toLowerCase();
      const value = decodeURIComponent(part.slice(idx + 1));
      if (!keyHints.some((h) => key.includes(h))) continue;
      const token = maybeTokenFromValue(value);
      if (token) return token;
    }
    return null;
  };

  const detectToken = () => {
    if (isLikelyToken(state.token)) return { token: state.token, source: "state.token" };

    const urlToken = extractTokenFromUrl(state.activeSocket?.url) || extractTokenFromUrl(state.lastWsUrl);
    if (isLikelyToken(urlToken)) return { token: urlToken, source: "wsUrl" };

    try {
      const perf = performance.getEntriesByType("resource") || [];
      for (let i = perf.length - 1; i >= 0; i -= 1) {
        const name = perf[i]?.name;
        if (!name || !name.includes(CFG.wsPath)) continue;
        const t = extractTokenFromUrl(name);
        if (isLikelyToken(t)) return { token: t, source: "performance.ws" };
      }
    } catch {}

    const lsToken = scanStorageForToken(window.localStorage);
    if (isLikelyToken(lsToken)) return { token: lsToken, source: "localStorage" };

    const ssToken = scanStorageForToken(window.sessionStorage);
    if (isLikelyToken(ssToken)) return { token: ssToken, source: "sessionStorage" };

    const cookieToken = scanCookieForToken();
    if (isLikelyToken(cookieToken)) return { token: cookieToken, source: "cookie(non-HttpOnly)" };

    return { token: null, source: null };
  };

  const upgradeMiningEfficiency = async (token, reason = "pre-weapon-upgrade") => {
    try {
      const res = await fetch(CFG.upgradeMiningEfficiencyPath, {
        method: "POST",
        headers: {
          accept: "*/*",
          "content-type": "application/json",
          authorization: `Bearer ${token}`
        },
        credentials: "include",
        body: "{}"
      });
      const text = await res.text();
      if (!res.ok) {
        warn(`[UPGRADE] Efficiency gagal (${res.status}) [${reason}]: ${text}`);
        return { ok: false, status: res.status, body: text };
      }
      log(`[UPGRADE] Efficiency berhasil [${reason}]: ${text}`);
      return { ok: true, status: res.status, body: text };
    } catch (err) {
      warn(`[UPGRADE] Efficiency network error [${reason}]:`, err);
      return { ok: false, status: null, body: String(err) };
    }
  };

  const upgradeBowAttack = async (token, reason = "pre-arena-start") => {
    try {
      const res = await fetch(CFG.upgradeBowAttackPath, {
        method: "POST",
        headers: {
          accept: "*/*",
          "content-type": "application/json",
          authorization: `Bearer ${token}`
        },
        credentials: "include",
        body: "{}"
      });
      const text = await res.text();
      if (!res.ok) {
        warn(`[UPGRADE] Bow attack gagal (${res.status}) [${reason}]: ${text}`);
        return { ok: false, status: res.status, body: text };
      }
      log(`[UPGRADE] Bow attack berhasil [${reason}]: ${text}`);
      return { ok: true, status: res.status, body: text };
    } catch (err) {
      warn(`[UPGRADE] Bow attack network error [${reason}]:`, err);
      return { ok: false, status: null, body: String(err) };
    }
  };

  const startArena = async ({ token, autoConnect = true } = {}) => {
    if (state.startInFlight) {
      warn("start() masih berjalan. Tunggu selesai dulu.");
      return state.lastStart;
    }

    const detected = detectToken();
    const finalToken = token || detected.token;
    if (!finalToken) {
      throw new Error("Token kosong. Pakai pvArena2.setToken('TOKEN') dulu.");
    }
    if (!token && detected.source) {
      log(`Token auto-detected from ${detected.source}`);
    }

    log(`Start request -> ${CFG.startPath} (timeout ${CFG.startTimeoutMs}ms)`);
    state.startInFlight = true;

    try {
      await ensureNoActiveArenaBeforeStart();
      log("[PREP] Upgrade efficiency sebelum upgrade weapon...");
      await upgradeMiningEfficiency(finalToken, "pre-weapon");
      log("[PREP] Upgrade bow attack sebelum arena start...");
      await upgradeBowAttack(finalToken, "pre-start");

      const controller = new AbortController();
      const timeoutId = setTimeout(() => controller.abort(), CFG.startTimeoutMs);
      let res;

      try {
        res = await fetch(CFG.startPath, {
          method: "POST",
          headers: {
            accept: "*/*",
            "content-type": "application/json",
            authorization: `Bearer ${finalToken}`
          },
          credentials: "include",
          body: "{}",
          signal: controller.signal
        });
      } catch (err) {
        const msg = err?.name === "AbortError"
          ? `Start timeout > ${CFG.startTimeoutMs}ms`
          : `Start fetch error: ${err?.message || String(err)}`;
        state.lastStart = {
          time: new Date().toISOString(),
          status: null,
          ok: false,
          error: msg
        };
        warn(msg);
        return state.lastStart;
      } finally {
        clearTimeout(timeoutId);
      }

      log("Start response status:", res.status);

      const text = await res.text();
      const data = safeJson(text);

      state.lastStart = {
        time: new Date().toISOString(),
        status: res.status,
        ok: res.ok,
        data
      };

      if (!res.ok || !data || typeof data !== "object" || !data.runId) {
        const isCooldown = data?.error === "cooldown" && typeof data?.cooldownUntil === "string";
        if (isCooldown) {
          const cooldownScheduled = scheduleCooldownRetry(finalToken, data.cooldownUntil);
          const result = {
            ...state.lastStart,
            cooldownScheduled,
            cooldownRetryAt: state.cooldownRetryAt
          };
          warn("Start cooldown:", result);
          return result;
        }
        warn("Start gagal / runId tidak ada:", state.lastStart);
        return state.lastStart;
      }

      clearCooldownRetry();
      state.token = finalToken;
      state.runId = data.runId;
      state.battlesLeft = Number.isFinite(Number(data.battlesLeft)) ? Number(data.battlesLeft) : null;
      state.maxBattlesPerDay = Number.isFinite(Number(data.maxBattlesPerDay)) ? Number(data.maxBattlesPerDay) : null;
      state.lastWsUrl = buildWsUrl(finalToken, data.runId);
      log("Start OK. runId:", state.runId);
      log("WSS URL:", state.lastWsUrl);
      if (typeof data.battlesLeft !== "undefined") {
        log("Battles left:", data.battlesLeft);
      }

      if (autoConnect) connect(state.lastWsUrl);
      return {
        ...state.lastStart,
        runId: state.runId,
        wsUrl: state.lastWsUrl
      };
    } finally {
      state.startInFlight = false;
    }
  };

  const connect = (wsUrl) => {
    const url = wsUrl || state.lastWsUrl || (state.token && state.runId ? buildWsUrl(state.token, state.runId) : null);
    if (!url) throw new Error("WSS URL kosong. Jalankan pvArena2.start() dulu.");

    if (state.activeSocket && state.activeSocket.readyState <= 1) {
      try {
        state.activeSocket.__pvArena2SkipRestart = true;
        state.activeSocket.close();
      } catch {}
    }

    const ws = new orig.WebSocket(url);
    ws.addEventListener("open", () => {
      attachArenaSocket(ws, "manual.connect");
      log("WSS open:", url);
      sendInitialPositionOnce();
      runLoop();
    });
    return ws;
  };

  const getAliveMonsters = () =>
    (state.latestArena?.monsters ?? [])
      .map((m, i) => ({ index: i, m }))
      .filter(({ m }) => Number(m.hp) > 0);

  const distanceToPlayer = (monster) => {
    if (!state.playerPos) return Number.POSITIVE_INFINITY;
    const mx = parseNumber(monster?.tileX);
    const my = parseNumber(monster?.tileY);
    if (mx === null || my === null) return Number.POSITIVE_INFINITY;
    return Math.abs(mx - state.playerPos.x) + Math.abs(my - state.playerPos.y);
  };

  const distanceBetween = (a, b) => {
    if (!a || !b) return Number.POSITIVE_INFINITY;
    return Math.abs(a.x - b.x) + Math.abs(a.y - b.y);
  };

  const samePos = (a, b) => Boolean(a && b && a.x === b.x && a.y === b.y);
  const getMonsterPos = (monster) => toPos(monster?.tileX, monster?.tileY);

  const countNearbyThreatAtPos = (pos, excludedIndex = null) => {
    if (!pos) return Number.POSITIVE_INFINITY;
    let count = 0;
    for (const { index, m } of getAliveMonsters()) {
      if (excludedIndex !== null && index === excludedIndex) continue;
      const mp = getMonsterPos(m);
      if (!mp) continue;
      if (distanceBetween(pos, mp) <= CFG.approachThreatRadius) count += 1;
    }
    return count;
  };

  const buildApproachCandidates = (targetPos) => {
    if (!targetPos) return [];
    const raw = [
      { x: targetPos.x + 1, y: targetPos.y + 1 },
      { x: targetPos.x, y: targetPos.y + 1 },
      { x: targetPos.x + 1, y: targetPos.y },
      { x: targetPos.x + 1, y: targetPos.y - 1 },
      { x: targetPos.x, y: targetPos.y - 1 },
      { x: targetPos.x - 1, y: targetPos.y },
      { x: targetPos.x - 1, y: targetPos.y - 1 },
      { x: targetPos.x - 1, y: targetPos.y + 1 },
      { x: targetPos.x, y: targetPos.y }
    ];
    if (state.playerPos && distanceBetween(state.playerPos, targetPos) <= 1) {
      raw.unshift({ x: state.playerPos.x, y: state.playerPos.y });
    }

    const seen = new Set();
    const unique = [];
    for (const p of raw) {
      const key = `${p.x},${p.y}`;
      if (seen.has(key)) continue;
      seen.add(key);
      unique.push(p);
    }
    return unique;
  };

  const countDirectionalThreatTowardOthers = (targetIndex, targetPos, approachPos) => {
    if (!targetPos || !approachPos) return 0;
    const approachVector = { x: approachPos.x - targetPos.x, y: approachPos.y - targetPos.y };
    if (approachVector.x === 0 && approachVector.y === 0) return 0;

    let risk = 0;
    for (const { index, m } of getAliveMonsters()) {
      if (index === targetIndex) continue;
      const mp = getMonsterPos(m);
      if (!mp) continue;

      const distToTarget = distanceBetween(targetPos, mp);
      if (!Number.isFinite(distToTarget) || distToTarget > CFG.approachDirectionalRadius) continue;

      const monsterVector = { x: mp.x - targetPos.x, y: mp.y - targetPos.y };
      const alignment = approachVector.x * monsterVector.x + approachVector.y * monsterVector.y;
      if (alignment <= 0) continue;

      const proximityWeight = CFG.approachDirectionalRadius - distToTarget + 1;
      risk += alignment * proximityWeight;
    }

    return risk;
  };

  const pickBestApproachForTarget = (targetIndex) => {
    const target = state.latestArena?.monsters?.[targetIndex];
    const targetPos = getMonsterPos(target);
    if (!targetPos) return null;

    const homePos = { x: CFG.initialPosition.gridX, y: CFG.initialPosition.gridY };
    const candidates = buildApproachCandidates(targetPos);
    if (!candidates.length) return null;

    let best = null;
    for (const pos of candidates) {
      const additionalThreat = countNearbyThreatAtPos(pos, targetIndex);
      const travel = distanceBetween(state.playerPos, pos);
      const targetDistance = distanceBetween(pos, targetPos);
      const homeDistance = distanceBetween(pos, homePos);
      const directionalRisk = countDirectionalThreatTowardOthers(targetIndex, targetPos, pos);
      const sameTilePenalty = targetDistance === 0 ? 250 : 0;
      const score = (
        additionalThreat * 1_000
        + directionalRisk * CFG.approachDirectionalPenalty
        + targetDistance * 100
        + sameTilePenalty
        + (Number.isFinite(travel) ? travel : 0) * 10
        + homeDistance
      );
      const candidate = { pos, additionalThreat, directionalRisk, targetDistance, score };
      if (!best || candidate.score < best.score) best = candidate;
    }
    return best;
  };

  const pickBestEngageTarget = (engageThreatLimit = CFG.maxAdditionalThreatForEngage) => {
    const alive = getAliveMonsters();
    if (!alive.length) return null;
    let best = null;

    for (const entry of alive) {
      const approach = pickBestApproachForTarget(entry.index);
      if (!approach) continue;
      const crowdPenalty = approach.additionalThreat > engageThreatLimit
        ? (approach.additionalThreat - engageThreatLimit) * 5_000
        : 0;
      const score = approach.score + crowdPenalty;
      const candidate = { ...entry, approach, score };
      if (!best || candidate.score < best.score) best = candidate;
    }

    return best;
  };

  const pickNearestAliveMonster = () => {
    const alive = getAliveMonsters();
    if (!alive.length) return null;
    if (!state.playerPos) return alive[0];
    alive.sort((a, b) => distanceToPlayer(a.m) - distanceToPlayer(b.m));
    return alive[0];
  };

  const pickNearestAliveMonsterExcluding = (excludedIndex) => {
    const alive = getAliveMonsters().filter((x) => x.index !== excludedIndex);
    if (!alive.length) return null;
    if (!state.playerPos) return alive[0];
    alive.sort((a, b) => distanceToPlayer(a.m) - distanceToPlayer(b.m));
    return alive[0];
  };

  const pickNearbyAliveMonster = (maxDistance, excludedIndex = null) => {
    const nearest = excludedIndex === null
      ? pickNearestAliveMonster()
      : pickNearestAliveMonsterExcluding(excludedIndex);
    if (!nearest) return null;
    const distance = distanceToPlayer(nearest.m);
    if (!Number.isFinite(distance) || distance > maxDistance) return null;
    return { ...nearest, distance };
  };

  const send = (obj) => {
    const ws = state.activeSocket;
    if (!ws || ws.readyState !== WebSocket.OPEN) return false;
    ws.send(JSON.stringify(obj));
    if (obj?.type === "position") {
      const p = toPos(obj.gridX, obj.gridY);
      if (p) state.playerPos = p;
    }
    return true;
  };

  const afterMonsterKilledTransition = async (killedIndex) => {
    const nearby = pickNearbyAliveMonster(CFG.chainAttackMaxDistance, killedIndex);
    if (nearby) {
      log(
        `Monster ${killedIndex} mati. Monster dekat ${nearby.index} distance=${nearby.distance} ` +
        `<= ${CFG.chainAttackMaxDistance}, lanjut chain attack sebelum cooldown.`
      );
      return;
    }

    send({ type: "position", gridX: CFG.initialPosition.gridX, gridY: CFG.initialPosition.gridY });
    log(`Monster ${killedIndex} mati. Return ke home (${CFG.initialPosition.gridX},${CFG.initialPosition.gridY}).`);

    const getPlayerHpRatio = () => {
      const arena = state.latestArena || {};
      const current = parseNumber(
        arena.playerHp
        ?? arena.playerCurrentHp
        ?? arena.currentHp
        ?? arena.character?.currentHp
      );
      const max = parseNumber(
        arena.playerMaxHp
        ?? arena.maxHp
        ?? arena.character?.maxHp
      );
      if (current === null || max === null || max <= 0) return null;
      return current / max;
    };

    const waitStartAt = Date.now();
    let prevHpRatio = getPlayerHpRatio();
    while (!state.stopped) {
      const ratio = getPlayerHpRatio();
      const closeThreat = pickNearbyAliveMonster(CFG.cooldownThreatDistance);
      const hpDropped = ratio !== null && prevHpRatio !== null && ratio + 0.0001 < prevHpRatio;
      if (closeThreat) {
        log(
          `[DEFENSE] Persiapan cooldown dibatalkan. Monster ${closeThreat.index} ` +
          `distance=${closeThreat.distance} <= ${CFG.cooldownThreatDistance}. Balas serang.`
        );
        return;
      }
      if (hpDropped) {
        const retaliation = pickNearestAliveMonster();
        if (retaliation) {
          log(`[DEFENSE] HP turun saat persiapan cooldown. Balas serang monster ${retaliation.index}.`);
          return;
        }
      }
      if (ratio === null) {
        warn("Data HP tidak tersedia, skip tunggu HP.");
        break;
      }
      if (ratio > CFG.hpResumeThresholdPct) {
        log(`HP ${(ratio * 100).toFixed(1)}% > ${(CFG.hpResumeThresholdPct * 100).toFixed(0)}%, lanjut cooldown.`);
        break;
      }
      if (Date.now() - waitStartAt >= CFG.hpRefillWaitTimeoutMs) {
        warn(
          `Timeout tunggu HP > ${(CFG.hpResumeThresholdPct * 100).toFixed(0)}% ` +
          `(${CFG.hpRefillWaitTimeoutMs}ms), lanjut cooldown.`
        );
        break;
      }
      prevHpRatio = ratio;
      await sleep(CFG.hpRefillPollMs);
    }

    if (state.stopped) return;
    log(`Cooldown antar monster: ${CFG.afterKillCooldownMs}ms`);
    const cooldownStartAt = Date.now();
    let cooldownPrevHpRatio = getPlayerHpRatio();
    while (!state.stopped && Date.now() - cooldownStartAt < CFG.afterKillCooldownMs) {
      const ratio = getPlayerHpRatio();
      const closeThreat = pickNearbyAliveMonster(CFG.cooldownThreatDistance);
      const hpDropped = ratio !== null && cooldownPrevHpRatio !== null && ratio + 0.0001 < cooldownPrevHpRatio;
      if (closeThreat) {
        log(
          `[DEFENSE] Cooldown dibatalkan. Monster ${closeThreat.index} ` +
          `distance=${closeThreat.distance} <= ${CFG.cooldownThreatDistance}. Balas serang.`
        );
        return;
      }
      if (hpDropped) {
        const retaliation = pickNearestAliveMonster();
        if (retaliation) {
          log(`[DEFENSE] HP turun saat cooldown. Balas serang monster ${retaliation.index}.`);
          return;
        }
      }
      if (ratio !== null) cooldownPrevHpRatio = ratio;
      const elapsed = Date.now() - cooldownStartAt;
      const remaining = Math.max(0, CFG.afterKillCooldownMs - elapsed);
      if (!remaining) break;
      await sleep(Math.min(CFG.hpRefillPollMs, remaining));
    }
  };

  const acquireSocket = async () => {
    const start = Date.now();
    while (!state.stopped && Date.now() - start < CFG.acquireTimeoutMs) {
      if (state.activeSocket && state.activeSocket.readyState === WebSocket.OPEN) {
        sendInitialPositionOnce();
        return true;
      }
      await sleep(250);
    }
    return false;
  };

  async function runLoop() {
    if (state.running || state.stopped) return;
    state.running = true;
    const loopSocketGeneration = state.activeSocket?.__pvArena2SocketGeneration ?? state.socketGeneration;

    try {
      const ok = await acquireSocket();
      if (!ok) {
        warn("Socket arena belum open. Jalankan pvArena2.start() atau pvArena2.connect().");
        return;
      }
      let noSafeEngageCycles = 0;

      while (!state.stopped && loopSocketGeneration === state.socketGeneration) {
        let engageThreatLimit = CFG.maxAdditionalThreatForEngage;
        let targetChoice = pickBestEngageTarget(engageThreatLimit);
        if (!targetChoice) {
          await sleep(120);
          continue;
        }

        if (targetChoice.approach.additionalThreat > engageThreatLimit) {
          noSafeEngageCycles += 1;
          const shouldFallback = noSafeEngageCycles >= CFG.safeEngageStallCycles;
          if (shouldFallback) {
            engageThreatLimit = Math.max(CFG.fallbackAdditionalThreatForEngage, CFG.maxAdditionalThreatForEngage);
            const relaxed = pickBestEngageTarget(engageThreatLimit);
            if (relaxed && relaxed.approach.additionalThreat <= engageThreatLimit) {
              targetChoice = relaxed;
              noSafeEngageCycles = 0;
              log(
                `[RISK] Anti-stall fallback: tidak ada target full-aman selama ${CFG.safeEngageStallCycles} siklus. ` +
                `Pakai threat limit ${engageThreatLimit} untuk target ${targetChoice.index}.`
              );
            }
          }
        } else {
          noSafeEngageCycles = 0;
        }

        if (targetChoice.approach.additionalThreat > engageThreatLimit) {
          send({ type: "position", gridX: CFG.initialPosition.gridX, gridY: CFG.initialPosition.gridY });
          log(
            `[RISK] Tahan engage. Target ${targetChoice.index} punya threat sekitar=${targetChoice.approach.additionalThreat} ` +
            `(max ${engageThreatLimit}). Reposition ke home.`
          );
          await sleep(CFG.crowdRepositionWaitMs);
          continue;
        }

        const targetIndex = targetChoice.index;
        while (!state.stopped && loopSocketGeneration === state.socketGeneration) {
          const target = state.latestArena?.monsters?.[targetIndex];
          if (!target || Number(target.hp) <= 0) {
            if (target && Number(target.hp) <= 0) {
              await afterMonsterKilledTransition(targetIndex);
            }
            break;
          }

          const targetPos = getMonsterPos(target);
          const approach = pickBestApproachForTarget(targetIndex);
          if (!targetPos || !approach) break;

          const currentTargetDistance = distanceBetween(state.playerPos, targetPos);
          const tooCrowdedToReposition = (
            approach.additionalThreat > engageThreatLimit
            && (!Number.isFinite(currentTargetDistance) || currentTargetDistance > 1)
          );
          if (tooCrowdedToReposition) break;

          if (!samePos(state.playerPos, approach.pos) && !send({
            type: "position",
            gridX: approach.pos.x,
            gridY: approach.pos.y
          })) {
            if (loopSocketGeneration !== state.socketGeneration) break;
            await sleep(300);
            continue;
          }
          await sleep(CFG.positionDelayMs);

          for (let n = 0; n < CFG.attackBurst; n += 1) {
            if (state.stopped) break;
            const check = state.latestArena?.monsters?.[targetIndex];
            if (!check || Number(check.hp) <= 0) break;
            if (!send({ type: "attack", monsterIndex: targetIndex })) break;
            await sleep(randInt(CFG.attackDelayMinMs, CFG.attackDelayMaxMs));
          }

          await sleep(CFG.betweenBurstMs);
        }
      }
    } finally {
      state.running = false;
    }
  }

  const stop = () => {
    state.stopped = true;
    clearCooldownRetry();
    log("Stopped.");
  };

  const resume = () => {
    if (!state.stopped) return;
    state.stopped = false;
    runLoop();
  };

  const autoStart = async () => {
    try {
      log("Auto start on load...");
      const result = await startArena({ autoConnect: true });
      if (!result?.ok) {
        warn("Auto start gagal:", result);
      }
      return result;
    } catch (e) {
      warn("Auto start error:", e?.message || e);
      return null;
    }
  };

  installHooks();

  window.pvArena2 = {
    setToken,
    setAutoDaily,
    detectToken,
    start: startArena,
    connect,
    autoStart,
    stop,
    resume,
    send,
    wsUrl: () => state.lastWsUrl,
    lastStart: () => state.lastStart,
    status: () => ({
      running: state.running,
      stopped: state.stopped,
      tokenSet: Boolean(state.token),
      runId: state.runId,
      battlesLeft: state.battlesLeft,
      maxBattlesPerDay: state.maxBattlesPerDay,
      playerPos: state.playerPos,
      autoDailyEnabled: state.autoDailyEnabled,
      autoRestarting: state.autoRestarting,
      autoRestartCount: state.autoRestartCount,
      startInFlight: state.startInFlight,
      awaitingActiveArena: state.awaitingActiveArena,
      cooldownRetryAt: state.cooldownRetryAt,
      cooldownRetryScheduled: Boolean(state.cooldownRetryTimer),
      activeSocketUrl: state.activeSocket?.url ?? null,
      activeSocketReadyState: state.activeSocket?.readyState ?? null,
      lastBattleOutcome: state.lastBattleOutcome
    }),
    state: () => state.latestArena
  };

  log("Ready.");
  log("Use:");
  log("1) script ini auto start saat load");
  log("2) pvArena2.setAutoDaily(false)  // matikan auto restart");
  log("3) pvArena2.status()");
  log("4) pvArena2.autoStart()  // trigger manual kalau perlu");

  if (CFG.autoStartOnLoad) {
    autoStart();
  }
})();
