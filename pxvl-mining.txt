let MODE = "normal"; // normal / fast
const MAP_ID = "map2";
const ROCK_MIN = 0;
const ROCK_MAX = 5;
const SELL_EVERY_SUCCESS = 25;
const LOW_CAPACITY_FREE_SLOT_TRIGGER = 4;
const ROCKS_SNAPSHOT_PATH = "/api/mining/rocks";
const ALL_ROCKS_EMPTY_WAIT_MS = 10 * 60 * 1000;
const AUTO_SELL_TIME_GATE_ENABLED = true;
const AUTO_SELL_BLOCKED_WINDOW_START = { hour: 7, minute: 0 };
const AUTO_SELL_BLOCKED_WINDOW_END = { hour: 9, minute: 0 };

let running = true;
let rockIndex = null;
let successCount = 0;
let nextSellAt = SELL_EVERY_SUCCESS;
let lastNearFullSellAttemptAt = 0;

function normalizeToken(raw) {
  if (!raw) return "";
  return String(raw).replace(/^Bearer\s+/i, "").trim();
}

function getToken() {
  return normalizeToken(
    localStorage.getItem("pixelvalley_token") ||
    localStorage.getItem("token") ||
    sessionStorage.getItem("pixelvalley_token")
  );
}

function getDelay() {
  if (MODE === "fast") {
    return Math.floor(Math.random() * 500) + 1000; // 1 - 1.5 detik
  }
  return Math.floor(Math.random() * 2001) + 3000; // 3 - 5 detik
}

function sellBlockedWindowLabel() {
  const start = `${String(AUTO_SELL_BLOCKED_WINDOW_START.hour).padStart(2, "0")}:${String(AUTO_SELL_BLOCKED_WINDOW_START.minute).padStart(2, "0")}`;
  const end = `${String(AUTO_SELL_BLOCKED_WINDOW_END.hour).padStart(2, "0")}:${String(AUTO_SELL_BLOCKED_WINDOW_END.minute).padStart(2, "0")}`;
  return `${start} - ${end}`;
}

function isWithinBlockedSellWindow(at = new Date()) {
  const start = new Date(at);
  start.setHours(AUTO_SELL_BLOCKED_WINDOW_START.hour, AUTO_SELL_BLOCKED_WINDOW_START.minute, 0, 0);
  const end = new Date(at);
  end.setHours(AUTO_SELL_BLOCKED_WINDOW_END.hour, AUTO_SELL_BLOCKED_WINDOW_END.minute, 0, 0);

  if (end.getTime() <= start.getTime()) {
    return at.getTime() >= start.getTime() || at.getTime() < end.getTime();
  }
  return at.getTime() >= start.getTime() && at.getTime() < end.getTime();
}

function isAutoSellTimeReached(at = new Date()) {
  if (!AUTO_SELL_TIME_GATE_ENABLED) return true;
  return !isWithinBlockedSellWindow(at);
}

function stop(reason) {
  console.log("[STOP]", reason);
  running = false;
}

function getCapacity(snapshot) {
  return Number(snapshot?.character?.capacity ?? snapshot?.capacity ?? 0);
}

function getLevel(snapshot) {
  return Number(snapshot?.character?.level ?? snapshot?.level ?? 0);
}

function getDailyMined(snapshot) {
  return Number(snapshot?.character?.dailyMinedToday ?? snapshot?.dailyMinedToday ?? 0);
}

function getDailyLimit(snapshot) {
  return Number(snapshot?.character?.dailyMiningLimit ?? snapshot?.dailyMiningLimit ?? 0);
}

function countFilledSlots(slots = []) {
  return slots.filter(s => s.resourceType !== "empty").length;
}

function getBagUsed(snapshot) {
  const slots = snapshot?.slots || [];
  return slots.reduce((sum, s) => {
    if (!s || s.resourceType === "empty") return sum;
    const count = Number(s.count || 0);
    return sum + (Number.isFinite(count) && count > 0 ? count : 0);
  }, 0);
}

function sumResources(slots = []) {
  const totals = { gold: 0, silver: 0, bronze: 0 };
  for (const s of slots) {
    if (Object.prototype.hasOwnProperty.call(totals, s.resourceType)) {
      totals[s.resourceType] += Number(s.count || 0);
    }
  }
  return totals;
}

async function fetchCharacter(token) {
  try {
    const res = await fetch("/api/character", {
      credentials: "include",
      headers: {
        "Authorization": `Bearer ${token}`
      }
    });
    if (!res.ok) return null;
    return await res.json();
  } catch {
    return null;
  }
}

function logCharacter(data) {
  if (!data) return;
  const totals = sumResources(data.slots || []);
  const capacity = getCapacity(data);
  const used = getBagUsed(data);
  const filledSlots = countFilledSlots(data.slots || []);
  console.log(
    `GOLD:${totals.gold} | SILVER:${totals.silver} | BRONZE:${totals.bronze} | Bag:${used}/${capacity} (slots:${filledSlots}) | Daily:${getDailyMined(data)}/${getDailyLimit(data)}`
  );
}

async function upgradeMiningEfficiency(token, reason = "unspecified") {
  try {
    const res = await fetch("/api/character/upgrade-mining-efficiency", {
      method: "POST",
      credentials: "include",
      headers: {
        accept: "*/*",
        "Content-Type": "application/json",
        "Authorization": `Bearer ${token}`
      },
      body: "{}"
    });

    const text = await res.text();
    if (!res.ok) {
      console.log(`[UPGRADE] Gagal (${res.status}) [${reason}]: ${text}`);
      return { ok: false, status: res.status, body: text };
    }

    console.log(`[UPGRADE] Berhasil [${reason}]: ${text}`);
    return { ok: true, status: res.status, body: text };
  } catch (err) {
    console.log(`[UPGRADE] Network error [${reason}]:`, err);
    return { ok: false, status: null, body: String(err) };
  }
}

async function sellSilverBronzeAll(token, slots) {
  // Prioritaskan data terbaru dari /api/character sebelum sell.
  const charData = await fetchCharacter(token);
  const source = charData || slots || {};
  const level = getLevel(source);
  if (level < 3) {
    console.log(`[SELL] Skip: level ${level} (sell unlock di level 3).`);
    return { ok: true, sold: false, skippedByLevel: true };
  }

  let totals = sumResources(charData?.slots || []);
  if (!totals.silver && !totals.bronze) {
    totals = sumResources(source?.slots || []);
  }

  if (!totals.silver && !totals.bronze) {
    console.log("[SELL] Tidak ada silver/bronze untuk dijual.");
    return { ok: true, sold: false };
  }

  const payload = {
    gold: 0,
    silver: totals.silver,
    bronze: totals.bronze
  };

  try {
    const res = await fetch("/api/market/sell", {
      method: "POST",
      credentials: "include",
      headers: {
        "Content-Type": "application/json",
        "Authorization": `Bearer ${token}`
      },
      body: JSON.stringify(payload)
    });

    const text = await res.text();
    if (!res.ok) {
      console.log(`[SELL] Gagal (${res.status}): ${text}`);
      return { ok: false, sold: false };
    }

    console.log(`[SELL] Berhasil jual silver=${payload.silver}, bronze=${payload.bronze}. Resp: ${text}`);
    await upgradeMiningEfficiency(token, "post-sell");
    return { ok: true, sold: true };
  } catch (err) {
    console.log("[SELL] Network error:", err);
    return { ok: false, sold: false };
  }
}

async function maybeAutoSell(token, snapshot) {
  if (!isAutoSellTimeReached()) return;

  const capacity = getCapacity(snapshot);
  const used = getBagUsed(snapshot);
  const freeCapacity = Math.max(capacity - used, 0);

  if (capacity > 50) {
    if (successCount < nextSellAt) return;
    console.log(`[SELL] Trigger: ${successCount} mining sukses (capacity ${capacity}). Menjual silver+bronze...`);
    const result = await sellSilverBronzeAll(token, snapshot);
    if (result.sold) {
      nextSellAt += SELL_EVERY_SUCCESS;
    } else if (result.skippedByLevel) {
      // Retry berkala tanpa spam tiap hit saat level masih < 3.
      nextSellAt = successCount + 5;
    }
    return;
  }

  if (freeCapacity > LOW_CAPACITY_FREE_SLOT_TRIGGER) return;
  if (successCount - lastNearFullSellAttemptAt < 3) return;

  lastNearFullSellAttemptAt = successCount;
  console.log(`[SELL] Trigger near-full (capacity ${capacity}, used ${used}, free ${freeCapacity}). Menjual silver+bronze...`);
  await sellSilverBronzeAll(token, snapshot);
}

function extractSnapshotFromMining(data) {
  return {
    character: data?.character || {},
    slots: data?.slots || [],
    dailyMinedToday: data?.dailyMinedToday,
    dailyMiningLimit: data?.dailyMiningLimit,
    capacity: data?.capacity
  };
}

function isBackpackFull(snapshot) {
  const capacity = getCapacity(snapshot);
  if (!capacity) return false;
  return getBagUsed(snapshot) >= capacity;
}

function isDailyLimitReached(snapshot) {
  const limit = getDailyLimit(snapshot);
  if (!limit) return false;
  return getDailyMined(snapshot) >= limit;
}

async function getLatestSnapshot(token, fallbackSnapshot) {
  const latest = await fetchCharacter(token);
  if (latest) return latest;
  return fallbackSnapshot;
}

async function fetchRocksSnapshot(token) {
  try {
    const url = `${ROCKS_SNAPSHOT_PATH}?mapId=${encodeURIComponent(MAP_ID)}`;
    const res = await fetch(url, {
      credentials: "include",
      headers: {
        accept: "*/*",
        "Authorization": `Bearer ${token}`
      }
    });
    const text = await res.text();
    if (!res.ok) {
      console.log(`[ROCK] Snapshot gagal (${res.status}): ${text}`);
      return null;
    }
    const data = safeJson(text);
    if (!data || typeof data !== "object" || !Array.isArray(data.rocks)) {
      console.log("[ROCK] Snapshot payload invalid:", data);
      return null;
    }
    return data.rocks;
  } catch (err) {
    console.log("[ROCK] Snapshot network error:", err);
    return null;
  }
}

async function pickRandomAliveRock(token, reason = "unspecified") {
  const rocks = await fetchRocksSnapshot(token);
  if (!rocks) return { ok: false, waitMs: 5000 };

  const available = rocks
    .filter((r) => {
      const idx = Number(r?.rockIndex);
      const hp = Number(r?.healthPct);
      return Number.isInteger(idx)
        && idx >= ROCK_MIN
        && idx <= ROCK_MAX
        && Number.isFinite(hp)
        && hp > 0;
    })
    .map((r) => ({ rockIndex: Number(r.rockIndex), healthPct: Number(r.healthPct) }));

  if (!available.length) {
    rockIndex = null;
    console.log(`[ROCK] Semua batu healthPct=0. Tunggu ${Math.ceil(ALL_ROCKS_EMPTY_WAIT_MS / 60000)} menit...`);
    return { ok: false, waitMs: ALL_ROCKS_EMPTY_WAIT_MS, allDepleted: true };
  }

  const picked = available[Math.floor(Math.random() * available.length)];
  rockIndex = picked.rockIndex;
  console.log(`[ROCK] Pilih random rock ${rockIndex} [${reason}] (healthPct=${picked.healthPct}).`);
  return { ok: true, rockIndex: picked.rockIndex, healthPct: picked.healthPct };
}

async function refreshCharacter(token) {
  const data = await fetchCharacter(token);
  if (!data) return null;
  logCharacter(data);
  return data;
}

async function evaluateLimitsOrStop(token, fallbackSnapshot) {
  const latest = await getLatestSnapshot(token, fallbackSnapshot);
  if (!latest) return;

  if (isDailyLimitReached(latest)) {
    stop("Daily limit reached");
    return;
  }

  if (isBackpackFull(latest)) {
    const level = getLevel(latest);
    if (level < 3) {
      stop("Backpack full while level < 3. Naikkan level via arena dulu.");
      return;
    }

    if (!isAutoSellTimeReached()) {
      stop(`Backpack full, tapi jam sell diblokir (${sellBlockedWindowLabel()}).`);
      return;
    }

    console.log("[BAG] Full dan level >= 3. Coba sell silver/bronze dulu...");
    const sellResult = await sellSilverBronzeAll(token, latest);
    if (!sellResult?.sold) {
      stop("Backpack full dan sell gagal. Stop mining.");
      return;
    }

    const afterSell = await fetchCharacter(token);
    if (afterSell && isBackpackFull(afterSell)) {
      stop("Backpack masih full setelah sell. Stop mining.");
      return;
    }

    if (afterSell) {
      logCharacter(afterSell);
      return;
    }
  }

  logCharacter(latest);
}

async function hitLoop() {
  if (!running) return;

  const token = getToken();
  if (!token) {
    stop("Token tidak ditemukan di storage.");
    return;
  }

  if (!Number.isInteger(rockIndex) || rockIndex < ROCK_MIN || rockIndex > ROCK_MAX) {
    const selected = await pickRandomAliveRock(token, "no_current_target");
    if (!selected.ok) {
      if (!running) return;
      setTimeout(hitLoop, selected.waitMs);
      return;
    }
  }

  try {
    const res = await fetch("/api/mining/hit", {
      method: "POST",
      credentials: "include",
      headers: {
        "Content-Type": "application/json",
        "Authorization": `Bearer ${token}`
      },
      body: JSON.stringify({
        mapId: MAP_ID,
        rockIndex: rockIndex
      })
    });

    if (!res.ok) {
      const raw = await res.text();
      let errBody = null;
      try { errBody = JSON.parse(raw); } catch {}

      if (errBody?.error === "daily_limit_reached") {
        stop("Daily limit reached (hit response)");
        return;
      }
      if (errBody?.error === "inventory_full") {
        console.log("[BAG] /api/mining/hit returned inventory_full. Checking backpack state...");
        await evaluateLimitsOrStop(token);
        if (!running) return;
        setTimeout(hitLoop, 1000);
        return;
      }
      if (errBody?.error === "rock_depleted") {
        console.log(`[ROCK] Rock ${rockIndex} already depleted (non-200 response). Pilih batu lain.`);
        rockIndex = null;
        setTimeout(hitLoop, 1000);
        return;
      }

      console.log(`[HIT] Rock ${rockIndex} tidak bisa di-hit (${res.status}) ${raw ? `| ${raw}` : ""}`);
      rockIndex = null;

      setTimeout(hitLoop, 1000);
      return;
    }

    const data = await res.json();
    if (data?.error === "daily_limit_reached") {
      stop("Daily limit reached (hit response)");
      return;
    }
    if (data?.error === "inventory_full") {
      console.log("[BAG] Mining response returned inventory_full. Checking backpack state...");
      await evaluateLimitsOrStop(token);
      if (!running) return;
      setTimeout(hitLoop, 1000);
      return;
    }
    if (data?.error === "rock_depleted") {
      console.log(`[ROCK] Rock ${rockIndex} depleted. Pilih batu lain.`);
      rockIndex = null;
      setTimeout(hitLoop, 1000);
      return;
    }

    successCount += 1;
    console.log(`[HIT] Sukses #${successCount} | Rock ${rockIndex} HP: ${data.rockHealthPct}`);

    if (data.drop && data.drop.resourceType !== "nothing") {
      console.log(`[DROP] +${data.drop.count} ${data.drop.resourceType}`);
    }

    const miningSnapshot = extractSnapshotFromMining(data);
    await maybeAutoSell(token, miningSnapshot);
    await evaluateLimitsOrStop(token, miningSnapshot);
    if (!running) return;

    if (data.rockHealthPct <= 0) {
      console.log(`[ROCK] Rock ${rockIndex} destroyed`);
      rockIndex = null;
    }

  } catch (err) {
    console.log("[ERR] Network error:", err);
  }

  setTimeout(hitLoop, getDelay());
}

async function startMining() {
  const token = getToken();
  if (!token) {
    stop("Token tidak ditemukan di storage.");
    return;
  }

  console.log("[START] Upgrade mining efficiency sebelum mining...");
  await upgradeMiningEfficiency(token, "pre-mining");

  console.log("[START] Pre-sell silver+bronze sebelum mining...");
  await sellSilverBronzeAll(token, null);
  await refreshCharacter(token);

  console.log("[START] Mining started (random rock from snapshot)");
  hitLoop();
}

startMining();
