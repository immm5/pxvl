  const sleep = (ms) => new Promise((resolve) => setTimeout(resolve, ms));

  const extractTokenFromUrl = (url) => {
    if (!url || typeof url !== "string") return null;
    try {
      const u = new URL(url, location.origin);
      const t = u.searchParams.get("token");
      return t || null;
    } catch {
      return null;
    }

  const normalizeToken = (raw) => {
    if (!raw) return "";
    return String(raw).replace(/^Bearer\s+/i, "").trim();
  };

  const isLikelyToken = (value) => {
    if (typeof value !== "string") return false;
    const v = value.trim();
    if (!v || v.length < 20) return false;
    if (/\s/.test(v)) return false;
    return true;
  };

  const maybeTokenFromValue = (value) => {
    const normalized = normalizeToken(value);
    return isLikelyToken(normalized) ? normalized : null;
  };

  const scanStorageForToken = (storage) => {
    if (!storage) return null;
    const keyHints = ["token", "auth", "access", "bearer", "jwt", "session", "pixelvalley"];
    const objHints = ["token", "accessToken", "authToken", "bearer", "authorization"];

    for (let i = 0; i < storage.length; i += 1) {
      const key = storage.key(i);
      if (!key) continue;

      let value = "";
      try {
        value = storage.getItem(key) ?? "";
      } catch {
        continue;
      }

      const direct = maybeTokenFromValue(value);
      if (direct && keyHints.some((h) => key.toLowerCase().includes(h))) {
        return direct;
      }

      if (value.startsWith("{") || value.startsWith("[")) {
        const parsed = safeJson(value);
        if (parsed && typeof parsed === "object") {
          for (const hint of objHints) {
            const nested = parsed?.[hint];
            const token = maybeTokenFromValue(typeof nested === "string" ? nested : "");
            if (token) return token;
          }
        }
      }
    }

    return null;
  };

  const scanCookieForToken = () => {
    let raw = "";
    try {
      raw = document.cookie || "";
    } catch {
      return null;
    }
    if (!raw) return null;

    const keyHints = ["token", "auth", "access", "bearer", "session"];
    const parts = raw.split(";").map((s) => s.trim()).filter(Boolean);
    for (const part of parts) {
      const idx = part.indexOf("=");
      if (idx < 1) continue;
      const key = decodeURIComponent(part.slice(0, idx)).toLowerCase();
      const value = decodeURIComponent(part.slice(idx + 1));
      if (!keyHints.some((h) => key.includes(h))) continue;
      const token = maybeTokenFromValue(value);
      if (token) return token;
    }

    return null;
  };

  const detectToken = ({ activeWsUrl = null, lastWsUrl = null } = {}) => {
    const urlToken = extractTokenFromUrl(activeWsUrl) || extractTokenFromUrl(lastWsUrl);
    if (isLikelyToken(urlToken)) return { token: urlToken, source: "wsUrl" };

    try {
      const perf = performance.getEntriesByType("resource") || [];
      for (let i = perf.length - 1; i >= 0; i -= 1) {
        const name = perf[i]?.name;
        if (!name) continue;
        const token = extractTokenFromUrl(name);
        if (isLikelyToken(token)) return { token, source: "performance" };
      }
    } catch {}

    const lsToken = scanStorageForToken(window.localStorage);
    if (isLikelyToken(lsToken)) return { token: lsToken, source: "localStorage" };

    const ssToken = scanStorageForToken(window.sessionStorage);
    if (isLikelyToken(ssToken)) return { token: ssToken, source: "sessionStorage" };

    const cookieToken = scanCookieForToken();
    if (isLikelyToken(cookieToken)) return { token: cookieToken, source: "cookie(non-HttpOnly)" };

    return { token: null, source: null };
  };

  function createArenaModule() {
    const cfg = CFG.arena;
    const wsOrigin = () => `${location.protocol === "https:" ? "wss:" : "ws:"}//${location.host}`;
    const log = (...args) => baseLog("[Arena]", ...args);
    const warn = (...args) => baseWarn("[Arena]", ...args);

    const state = {
      stopped: false,
      runningLoop: false,
      activeSocket: null,
      latestArena: null,
      playerPos: null,
      token: null,
      runId: null,
      lastWsUrl: null,
      lastStart: null,
      battlesLeft: null,
      maxBattlesPerDay: null,
      autoRestarting: false,
      autoRestartCount: 0,
      startInFlight: false,
      dailyInProgress: false,
      dailyPromise: null,
      dailyResolve: null,
      cooldownRetryTimer: null,
      cooldownRetryAt: null,
      awaitingActiveArena: false,
      lastBattleOutcome: null

    const parseNumber = (value) => {
      const n = Number(value);
      return Number.isFinite(n) ? n : null;
    };

    const setBattleOutcomeFromMessage = (data) => {
      if (data?.victory === true) {
        state.lastBattleOutcome = {
          type: "victory",
          playerHp: parseNumber(data.playerHp ?? state.latestArena?.playerHp),
          xpGained: parseNumber(data.xpGained),
          cooldownUntil: typeof data?.cooldownUntil === "string" ? data.cooldownUntil : null,
          at: new Date().toISOString()
        };
        return;
      }

      if (data?.defeat === true) {
        state.lastBattleOutcome = {
          type: "defeat",
          playerHp: parseNumber(data.playerHp ?? state.latestArena?.playerHp),
          xpGained: parseNumber(data.xpGained),
          cooldownUntil: typeof data?.cooldownUntil === "string" ? data.cooldownUntil : null,
          at: new Date().toISOString()
        };
        return;
      }

      if (data?.error === "cooldown" && typeof data?.cooldownUntil === "string") {
        if (!state.lastBattleOutcome || state.lastBattleOutcome.type === "cooldown") {
          state.lastBattleOutcome = {
            type: "cooldown",
            playerHp: parseNumber(data.playerHp ?? state.latestArena?.playerHp),
            xpGained: null,
            cooldownUntil: data.cooldownUntil,
            at: new Date().toISOString()
          };
        }
      }
    };

    const formatBattleOutcomeForLog = (outcome) => {
      if (!outcome) return "result=UNKNOWN";
      const parts = [];
      if (outcome.type === "victory") parts.push("result=WIN");
      else if (outcome.type === "defeat") parts.push("result=LOSE");
      else parts.push("result=COOLDOWN");

      if (Number.isFinite(outcome.playerHp)) parts.push(`playerHp=${outcome.playerHp}`);
      if (Number.isFinite(outcome.xpGained)) parts.push(`xp=${outcome.xpGained}`);
      if (outcome.cooldownUntil) parts.push(`cooldownUntil=${outcome.cooldownUntil}`);
      return parts.join(" | ");
    };

    const isArenaUrl = (url) => {
      if (typeof url !== "string") return false;
      const lower = url.toLowerCase();
      return lower.startsWith("ws")
        && lower.includes(cfg.wsPath.toLowerCase())
        && lower.includes(cfg.wsHostPattern.toLowerCase());
    };

    const toPos = (x, y) => {
      const px = parseNumber(x);
      const py = parseNumber(y);
      return px === null || py === null ? null : { x: px, y: py };
    };

    const extractPlayerPosFromState = (data) => {
      const pairs = [
        ["playerTileX", "playerTileY"],
        ["playerGridX", "playerGridY"],
        ["playerX", "playerY"],
        ["gridX", "gridY"],
        ["tileX", "tileY"]
      ];

      for (const [kx, ky] of pairs) {
        const p = toPos(data?.[kx], data?.[ky]);
        if (p) return p;
      }

      const nested = [data?.player, data?.position, data?.character];
      for (const obj of nested) {
        if (!obj || typeof obj !== "object") continue;
        const p1 = toPos(obj.tileX, obj.tileY);
        const p2 = toPos(obj.gridX, obj.gridY);
        const p3 = toPos(obj.x, obj.y);
        if (p1) return p1;
        if (p2) return p2;
        if (p3) return p3;
      }

      return null;
    };

    const updatePlayerPosFromState = (data) => {
      const pos = extractPlayerPosFromState(data);
      if (pos) state.playerPos = pos;
    };

    const parseIncoming = (raw) => {
      const data = safeJson(typeof raw === "string" ? raw : String(raw));
      if (!data || typeof data !== "object") return;
      setBattleOutcomeFromMessage(data);
      if (Array.isArray(data.monsters)) {
        state.latestArena = data;
        updatePlayerPosFromState(data);
        runLoop();
      }
    };

    const send = (obj) => {
      const ws = state.activeSocket;
      if (!ws || ws.readyState !== WebSocket.OPEN) return false;
      ws.send(JSON.stringify(obj));
      if (obj?.type === "position") {
        const p = toPos(obj.gridX, obj.gridY);
        if (p) state.playerPos = p;
      }
      return true;
    };

    const sendInitialPositionOnce = () => {
      const ws = state.activeSocket;
      if (!ws || ws.readyState !== WebSocket.OPEN) return false;
      if (ws.__pvDailyInitPosSent) return true;

      const initPos = {
        type: "position",
        gridX: cfg.initialPosition.gridX,
        gridY: cfg.initialPosition.gridY
      };

      try {
        ws.send(JSON.stringify(initPos));
        ws.__pvDailyInitPosSent = true;
        state.playerPos = { x: initPos.gridX, y: initPos.gridY };
        log("Initial position sent once:", initPos);
        return true;
      } catch (e) {
        warn("Failed to send initial position:", e);
        return false;
      }
    };

    const getAliveMonsters = () =>
      (state.latestArena?.monsters ?? [])
        .map((m, i) => ({ index: i, m }))
        .filter(({ m }) => Number(m.hp) > 0);

    const distanceToPlayer = (monster) => {
      if (!state.playerPos) return Number.POSITIVE_INFINITY;
      const mx = parseNumber(monster?.tileX);
      const my = parseNumber(monster?.tileY);
      if (mx === null || my === null) return Number.POSITIVE_INFINITY;
      return Math.abs(mx - state.playerPos.x) + Math.abs(my - state.playerPos.y);
    };

    const distanceBetween = (a, b) => {
      if (!a || !b) return Number.POSITIVE_INFINITY;
      return Math.abs(a.x - b.x) + Math.abs(a.y - b.y);
    };

    const samePos = (a, b) => Boolean(a && b && a.x === b.x && a.y === b.y);
    const getMonsterPos = (monster) => toPos(monster?.tileX, monster?.tileY);

    const countNearbyThreatAtPos = (pos, excludedIndex = null) => {
      if (!pos) return Number.POSITIVE_INFINITY;
      let count = 0;
      for (const { index, m } of getAliveMonsters()) {
        if (excludedIndex !== null && index === excludedIndex) continue;
        const mp = getMonsterPos(m);
        if (!mp) continue;
        if (distanceBetween(pos, mp) <= cfg.approachThreatRadius) count += 1;
      }
      return count;
    };

    const buildApproachCandidates = (targetPos) => {
      if (!targetPos) return [];
      const raw = [
        { x: targetPos.x + 1, y: targetPos.y + 1 },
        { x: targetPos.x, y: targetPos.y + 1 },
        { x: targetPos.x + 1, y: targetPos.y },
        { x: targetPos.x + 1, y: targetPos.y - 1 },
        { x: targetPos.x, y: targetPos.y - 1 },
        { x: targetPos.x - 1, y: targetPos.y },
        { x: targetPos.x - 1, y: targetPos.y - 1 },
        { x: targetPos.x - 1, y: targetPos.y + 1 },
        { x: targetPos.x, y: targetPos.y }
      ];
      if (state.playerPos && distanceBetween(state.playerPos, targetPos) <= 1) {
        raw.unshift({ x: state.playerPos.x, y: state.playerPos.y });
      }

      const seen = new Set();
      const unique = [];
      for (const p of raw) {
        const key = `${p.x},${p.y}`;
        if (seen.has(key)) continue;
        seen.add(key);
        unique.push(p);
      }
      return unique;
    };

    const countDirectionalThreatTowardOthers = (targetIndex, targetPos, approachPos) => {
      if (!targetPos || !approachPos) return 0;
      const approachVector = { x: approachPos.x - targetPos.x, y: approachPos.y - targetPos.y };
      if (approachVector.x === 0 && approachVector.y === 0) return 0;

      let risk = 0;
      for (const { index, m } of getAliveMonsters()) {
        if (index === targetIndex) continue;
        const mp = getMonsterPos(m);
        if (!mp) continue;

        const distToTarget = distanceBetween(targetPos, mp);
        if (!Number.isFinite(distToTarget) || distToTarget > cfg.approachDirectionalRadius) continue;

        const monsterVector = { x: mp.x - targetPos.x, y: mp.y - targetPos.y };
        const alignment = approachVector.x * monsterVector.x + approachVector.y * monsterVector.y;
        if (alignment <= 0) continue;

        const proximityWeight = cfg.approachDirectionalRadius - distToTarget + 1;
        risk += alignment * proximityWeight;
      }

      return risk;
    };

    const pickBestApproachForTarget = (targetIndex) => {
      const target = state.latestArena?.monsters?.[targetIndex];
      const targetPos = getMonsterPos(target);
      if (!targetPos) return null;

      const homePos = { x: cfg.initialPosition.gridX, y: cfg.initialPosition.gridY };
      const candidates = buildApproachCandidates(targetPos);
      if (!candidates.length) return null;

      let best = null;
      for (const pos of candidates) {
        const additionalThreat = countNearbyThreatAtPos(pos, targetIndex);
        const travel = distanceBetween(state.playerPos, pos);
        const targetDistance = distanceBetween(pos, targetPos);
        const homeDistance = distanceBetween(pos, homePos);
        const directionalRisk = countDirectionalThreatTowardOthers(targetIndex, targetPos, pos);
        const sameTilePenalty = targetDistance === 0 ? 250 : 0;
        const score = (
          additionalThreat * 1_000
          + directionalRisk * cfg.approachDirectionalPenalty
          + targetDistance * 100
          + sameTilePenalty
          + (Number.isFinite(travel) ? travel : 0) * 10
          + homeDistance
        );
        const candidate = { pos, additionalThreat, directionalRisk, targetDistance, score };
        if (!best || candidate.score < best.score) best = candidate;
      }
      return best;
    };

    const pickBestEngageTarget = (engageThreatLimit = cfg.maxAdditionalThreatForEngage) => {
      const alive = getAliveMonsters();
      if (!alive.length) return null;
      let best = null;

      for (const entry of alive) {
        const approach = pickBestApproachForTarget(entry.index);
        if (!approach) continue;
        const crowdPenalty = approach.additionalThreat > engageThreatLimit
          ? (approach.additionalThreat - engageThreatLimit) * 5_000
          : 0;
        const score = approach.score + crowdPenalty;
        const candidate = { ...entry, approach, score };
        if (!best || candidate.score < best.score) best = candidate;
      }

      return best;
    };

    const pickNearestAliveMonster = () => {
      const alive = getAliveMonsters();
      if (!alive.length) return null;
      if (!state.playerPos) return alive[0];
      alive.sort((a, b) => distanceToPlayer(a.m) - distanceToPlayer(b.m));
      return alive[0];
    };

    const pickNearestAliveMonsterExcluding = (excludedIndex) => {
      const alive = getAliveMonsters().filter((x) => x.index !== excludedIndex);
      if (!alive.length) return null;
      if (!state.playerPos) return alive[0];
      alive.sort((a, b) => distanceToPlayer(a.m) - distanceToPlayer(b.m));
      return alive[0];
    };

    const pickNearbyAliveMonster = (maxDistance, excludedIndex = null) => {
      const nearest = excludedIndex === null
        ? pickNearestAliveMonster()
        : pickNearestAliveMonsterExcluding(excludedIndex);
      if (!nearest) return null;
      const distance = distanceToPlayer(nearest.m);
      if (!Number.isFinite(distance) || distance > maxDistance) return null;
      return { ...nearest, distance };
    };

    const acquireSocket = async () => {
      const startAt = Date.now();
      while (!state.stopped && Date.now() - startAt < cfg.acquireTimeoutMs) {
        if (state.activeSocket && state.activeSocket.readyState === WebSocket.OPEN) {
          sendInitialPositionOnce();
          return true;
        }
        await sleep(250);
      }
      return false;
    };

    const afterMonsterKilledTransition = async (killedIndex) => {
      const nearby = pickNearbyAliveMonster(cfg.chainAttackMaxDistance, killedIndex);
      if (nearby) {
        log(
          `Monster ${killedIndex} dead. Nearby monster ${nearby.index} distance=${nearby.distance} ` +
          `<= ${cfg.chainAttackMaxDistance}, chain attack before cooldown.`
        );
        return;
      }

      send({ type: "position", gridX: cfg.initialPosition.gridX, gridY: cfg.initialPosition.gridY });
      log(`Monster ${killedIndex} dead. Return home (${cfg.initialPosition.gridX},${cfg.initialPosition.gridY}).`);

      const getPlayerHpRatio = () => {
        const arena = state.latestArena || {};
        const current = parseNumber(
          arena.playerHp
          ?? arena.playerCurrentHp
          ?? arena.currentHp
          ?? arena.character?.currentHp
        );
        const max = parseNumber(
          arena.playerMaxHp
          ?? arena.maxHp
          ?? arena.character?.maxHp
        );
        if (current === null || max === null || max <= 0) return null;
        return current / max;
      };

      const waitStartAt = Date.now();
      let prevHpRatio = getPlayerHpRatio();
      while (!state.stopped) {
        const ratio = getPlayerHpRatio();
        const closeThreat = pickNearbyAliveMonster(cfg.cooldownThreatDistance);
        const hpDropped = ratio !== null && prevHpRatio !== null && ratio + 0.0001 < prevHpRatio;
        if (closeThreat) {
          log(
            `[DEFENSE] Cooldown prep interrupted. Monster ${closeThreat.index} ` +
            `distance=${closeThreat.distance} <= ${cfg.cooldownThreatDistance}. Retaliating now.`
          );
          return;
        }
        if (hpDropped) {
          const retaliation = pickNearestAliveMonster();
          if (retaliation) {
            log(`[DEFENSE] HP dropped during cooldown prep. Retaliate monster ${retaliation.index}.`);
            return;
          }
        }
        if (ratio === null) {
          warn("HP data unavailable; skip HP wait.");
          break;
        }
        if (ratio > cfg.hpResumeThresholdPct) {
          log(`HP ${(ratio * 100).toFixed(1)}% > ${(cfg.hpResumeThresholdPct * 100).toFixed(0)}%, continue cooldown.`);
          break;
        }
        if (Date.now() - waitStartAt >= cfg.hpRefillWaitTimeoutMs) {
          warn(
            `Timeout waiting HP > ${(cfg.hpResumeThresholdPct * 100).toFixed(0)}% ` +
            `(${cfg.hpRefillWaitTimeoutMs}ms), continue cooldown.`
          );
          break;
        }
        prevHpRatio = ratio;
        await sleep(cfg.hpRefillPollMs);
      }

      if (state.stopped) return;
      log(`Cooldown between monsters: ${cfg.afterKillCooldownMs}ms`);
      const cooldownStartAt = Date.now();
      let cooldownPrevHpRatio = getPlayerHpRatio();
      while (!state.stopped && Date.now() - cooldownStartAt < cfg.afterKillCooldownMs) {
        const ratio = getPlayerHpRatio();
        const closeThreat = pickNearbyAliveMonster(cfg.cooldownThreatDistance);
        const hpDropped = ratio !== null && cooldownPrevHpRatio !== null && ratio + 0.0001 < cooldownPrevHpRatio;
        if (closeThreat) {
          log(
            `[DEFENSE] Cooldown interrupted. Monster ${closeThreat.index} ` +
            `distance=${closeThreat.distance} <= ${cfg.cooldownThreatDistance}. Retaliating now.`
          );
          return;
        }
        if (hpDropped) {
          const retaliation = pickNearestAliveMonster();
          if (retaliation) {
            log(`[DEFENSE] HP dropped during cooldown. Retaliate monster ${retaliation.index}.`);
            return;
          }
        }
        if (ratio !== null) cooldownPrevHpRatio = ratio;
        const elapsed = Date.now() - cooldownStartAt;
        const remaining = Math.max(0, cfg.afterKillCooldownMs - elapsed);
        if (!remaining) break;
        await sleep(Math.min(cfg.hpRefillPollMs, remaining));
      }
    };

    async function runLoop() {
      if (state.runningLoop || state.stopped) return;
      state.runningLoop = true;
      const loopSocket = state.activeSocket;

      try {
        const ok = await acquireSocket();
        if (!ok) return;
        let noSafeEngageCycles = 0;

        while (!state.stopped && loopSocket === state.activeSocket && loopSocket?.readyState === WebSocket.OPEN) {
          let engageThreatLimit = cfg.maxAdditionalThreatForEngage;
          let targetChoice = pickBestEngageTarget(engageThreatLimit);
          if (!targetChoice) {
            await sleep(120);
            continue;
          }

          if (targetChoice.approach.additionalThreat > engageThreatLimit) {
            noSafeEngageCycles += 1;
            const shouldFallback = noSafeEngageCycles >= cfg.safeEngageStallCycles;
            if (shouldFallback) {
              engageThreatLimit = Math.max(cfg.fallbackAdditionalThreatForEngage, cfg.maxAdditionalThreatForEngage);
              const relaxed = pickBestEngageTarget(engageThreatLimit);
              if (relaxed && relaxed.approach.additionalThreat <= engageThreatLimit) {
                targetChoice = relaxed;
                noSafeEngageCycles = 0;
                log(
                  `[RISK] Anti-stall fallback: no fully-safe target for ${cfg.safeEngageStallCycles} cycles. ` +
                  `Using threat limit ${engageThreatLimit} on target ${targetChoice.index}.`
                );
              }
            }
          } else {
            noSafeEngageCycles = 0;
          }

          if (targetChoice.approach.additionalThreat > engageThreatLimit) {
            send({ type: "position", gridX: cfg.initialPosition.gridX, gridY: cfg.initialPosition.gridY });
            log(
              `[RISK] Hold engage. Target ${targetChoice.index} has nearby threats=${targetChoice.approach.additionalThreat} ` +
              `(max ${engageThreatLimit}). Reposition to home.`
            );
            await sleep(cfg.crowdRepositionWaitMs);
            continue;
          }

          const targetIndex = targetChoice.index;
          while (!state.stopped && loopSocket === state.activeSocket && loopSocket?.readyState === WebSocket.OPEN) {
            const target = state.latestArena?.monsters?.[targetIndex];
            if (!target || Number(target.hp) <= 0) {
              if (target && Number(target.hp) <= 0) {
                await afterMonsterKilledTransition(targetIndex);
              }
              break;
            }

            const targetPos = getMonsterPos(target);
            const approach = pickBestApproachForTarget(targetIndex);
            if (!targetPos || !approach) break;

            const currentTargetDistance = distanceBetween(state.playerPos, targetPos);
            const tooCrowdedToReposition = (
              approach.additionalThreat > engageThreatLimit
              && (!Number.isFinite(currentTargetDistance) || currentTargetDistance > 1)
            );
            if (tooCrowdedToReposition) break;

            if (!samePos(state.playerPos, approach.pos) && !send({
              type: "position",
              gridX: approach.pos.x,
              gridY: approach.pos.y
            })) {
              await sleep(300);
              continue;
            }
            await sleep(cfg.positionDelayMs);

            for (let n = 0; n < cfg.attackBurst; n += 1) {
              if (state.stopped) break;
              const check = state.latestArena?.monsters?.[targetIndex];
              if (!check || Number(check.hp) <= 0) break;
              if (!send({ type: "attack", monsterIndex: targetIndex })) break;
              await sleep(randInt(cfg.attackDelayMinMs, cfg.attackDelayMaxMs));
            }

            await sleep(cfg.betweenBurstMs);
          }
        }
      } finally {
        state.runningLoop = false;
      }
    }

    const buildWsUrl = (token, runId) =>
      `${wsOrigin()}${cfg.wsPath}?token=${encodeURIComponent(token)}&runId=${encodeURIComponent(runId)}`;

    const findArenaUrlFromPerformance = () => {
      try {
        const entries = performance.getEntriesByType("resource") || [];
        for (let i = entries.length - 1; i >= 0; i -= 1) {
          const name = entries[i]?.name;
          if (isArenaUrl(name)) return name;
        }
      } catch {}
      return null;
    };

    const waitForSocketOpen = (ws, timeoutMs) =>
      new Promise((resolve) => {
        if (!ws) return resolve(false);
        if (ws.readyState === WebSocket.OPEN) return resolve(true);
        if (ws.readyState >= WebSocket.CLOSING) return resolve(false);

        let settled = false;
        const done = (ok) => {
          if (settled) return;
          settled = true;
          clearTimeout(timer);
          resolve(ok);
        };
        const onOpen = () => done(true);
        const onClose = () => done(false);
        const onError = () => done(false);
        const timer = setTimeout(() => done(false), timeoutMs);
        ws.addEventListener("open", onOpen, { once: true });
        ws.addEventListener("close", onClose, { once: true });
        ws.addEventListener("error", onError, { once: true });
      });

    const waitForSocketClose = (ws, timeoutMs) =>
      new Promise((resolve) => {
        if (!ws || ws.readyState >= WebSocket.CLOSING) return resolve(true);
        let settled = false;
        const done = (ok) => {
          if (settled) return;
          settled = true;
          clearTimeout(timer);
          resolve(ok);
        };
        const timer = setTimeout(() => done(false), timeoutMs);
        ws.addEventListener("close", () => done(true), { once: true });
      });

    const ensureNoActiveArenaBeforeStart = async () => {
      state.awaitingActiveArena = true;
      try {
        let ws = state.activeSocket;
        if (ws && ws.readyState === WebSocket.OPEN) {
          ws.__pvDailySkipRestart = true;
          log("Active arena detected. Finishing it before new start...");
          runLoop();
          const closed = await waitForSocketClose(ws, cfg.activeArenaWaitTimeoutMs);
          if (!closed) warn("Timeout waiting active arena to finish. Continue with fresh start.");
          return;
        }

        const perfUrl = findArenaUrlFromPerformance();
        if (!perfUrl) return;
        if (state.activeSocket?.url === perfUrl && state.activeSocket?.readyState === WebSocket.OPEN) return;

        log("Trying to attach existing arena from performance WSS...");
        ws = new WebSocket(perfUrl);
        ws.__pvDailySkipRestart = true;
        ws.addEventListener("open", () => {
          state.activeSocket = ws;
          state.lastBattleOutcome = null;
          log("WSS attached from performance:", perfUrl);
          sendInitialPositionOnce();
          runLoop();
        });
        ws.addEventListener("message", (ev) => parseIncoming(ev.data));
        ws.addEventListener("close", (ev) => handleSocketClose(ev, ws));
        ws.addEventListener("error", (ev) => warn("WSS attach error:", ev));

        const opened = await waitForSocketOpen(ws, cfg.preStartAttachTimeoutMs);
        if (!opened) {
          try { ws.close(); } catch {}
          return;
        }

        const closed = await waitForSocketClose(ws, cfg.activeArenaWaitTimeoutMs);
        if (!closed) warn("Timeout waiting performance-attached arena to finish. Continue with fresh start.");
      } finally {
        state.awaitingActiveArena = false;
      }
    };

    const clearCooldownRetry = () => {
      if (state.cooldownRetryTimer) {
        clearTimeout(state.cooldownRetryTimer);
        state.cooldownRetryTimer = null;
      }
      state.cooldownRetryAt = null;
    };

    const scheduleCooldownRetry = (token, cooldownUntil) => {
      const ts = Date.parse(cooldownUntil || "");
      if (!Number.isFinite(ts)) {
        warn("Cooldown received but cooldownUntil invalid:", cooldownUntil);
        return false;
      }

      const waitMs = Math.max(0, ts - Date.now()) + cfg.cooldownExtraWaitMs;
      clearCooldownRetry();
      state.cooldownRetryAt = new Date(Date.now() + waitMs).toISOString();

      log(
        `Arena cooldown active. Retry in ${Math.ceil(waitMs / 1000)}s ` +
        `(cooldownUntil + ${cfg.cooldownExtraWaitMs / 1000}s) at ${state.cooldownRetryAt}`
      );

      state.cooldownRetryTimer = setTimeout(async () => {
        state.cooldownRetryTimer = null;
        state.cooldownRetryAt = null;

        if (!state.dailyInProgress || state.stopped) return;

        try {
          const result = await startArena({ token, autoConnect: true });
          if (!result?.ok && !result?.cooldownScheduled) {
            const resultLeft = Number(result?.data?.battlesLeft);
            if (Number.isFinite(resultLeft) && resultLeft <= 0) {
              finishDaily("daily_limit_reached");
            } else {
              finishDaily("start_failed", result);
            }
          }
        } catch (e) {
          finishDaily("restart_error", { message: e?.message || String(e) });
        }
      }, waitMs);

      return true;
    };

    const upgradeMiningEfficiency = async (token, reason = "pre-weapon-upgrade") => {
      try {
        const res = await fetch(cfg.upgradeMiningEfficiencyPath, {
          method: "POST",
          headers: {
            accept: "*/*",
            "content-type": "application/json",
            authorization: `Bearer ${token}`
          },
          credentials: "include",
          body: "{}"
        });
        const text = await res.text();
        if (!res.ok) {
          warn(`[UPGRADE] Efficiency failed (${res.status}) [${reason}]: ${text}`);
          return { ok: false, status: res.status, body: text };
        }
        log(`[UPGRADE] Efficiency success [${reason}]: ${text}`);
        return { ok: true, status: res.status, body: text };
      } catch (err) {
        warn(`[UPGRADE] Efficiency network error [${reason}]:`, err);
        return { ok: false, status: null, body: String(err) };
      }
    };

    const upgradeBowAttack = async (token, reason = "pre-arena-start") => {
      try {
        const res = await fetch(cfg.upgradeBowAttackPath, {
          method: "POST",
          headers: {
            accept: "*/*",
            "content-type": "application/json",
            authorization: `Bearer ${token}`
          },
          credentials: "include",
          body: "{}"
        });
        const text = await res.text();
        if (!res.ok) {
          warn(`[UPGRADE] Bow attack failed (${res.status}) [${reason}]: ${text}`);
          return { ok: false, status: res.status, body: text };
        }
        log(`[UPGRADE] Bow attack success [${reason}]: ${text}`);
        return { ok: true, status: res.status, body: text };
      } catch (err) {
        warn(`[UPGRADE] Bow attack network error [${reason}]:`, err);
        return { ok: false, status: null, body: String(err) };
      }
    };

    const finishDaily = (reason, extra = null) => {
      if (!state.dailyInProgress) return;
      state.dailyInProgress = false;
      state.autoRestarting = false;
      clearCooldownRetry();
      log("Arena daily finished:", reason);
      if (typeof state.dailyResolve === "function") {
        state.dailyResolve({
          ok: true,
          reason,
          battlesLeft: state.battlesLeft,
          maxBattlesPerDay: state.maxBattlesPerDay,
          extra
        });
      }
      state.dailyResolve = null;
      state.dailyPromise = null;
    };

    const maybeAutoRestartAfterClose = async () => {
      if (!state.dailyInProgress) return;
      if (state.stopped) {
        finishDaily("stopped");
        return;
      }
      if (state.autoRestarting || state.startInFlight) return;

      const left = Number(state.battlesLeft);
      if (Number.isFinite(left) && left <= 0) {
        finishDaily("daily_limit_reached");
        return;
      }

      if (state.autoRestartCount >= cfg.maxAutoRestarts) {
        finishDaily("max_auto_restart_reached");
        return;
      }

      state.autoRestarting = true;
      state.autoRestartCount += 1;
      const waitMs = randInt(cfg.restartDelayMinMs, cfg.restartDelayMaxMs);
      log(`Next arena start in ${waitMs}ms (restart #${state.autoRestartCount})`);
      await sleep(waitMs);

      try {
        if (!state.dailyInProgress || state.stopped) return;
        const result = await startArena({ autoConnect: true });
        if (!result?.ok) {
          if (result?.cooldownScheduled) {
            log("Start hit cooldown. Retry already scheduled at:", result.cooldownRetryAt);
          } else {
            const resultLeft = Number(result?.data?.battlesLeft);
            if (Number.isFinite(resultLeft) && resultLeft <= 0) {
              finishDaily("daily_limit_reached");
            } else {
              finishDaily("start_failed", result);
            }
          }
        }
      } catch (e) {
        finishDaily("restart_error", { message: e?.message || String(e) });
      } finally {
        state.autoRestarting = false;
      }
    };

    const handleSocketClose = (ev, ws) => {
      if (state.activeSocket === ws) state.activeSocket = null;
      state.runningLoop = false;
      log(
        "WSS close:",
        ev?.code ?? "?",
        ev?.reason || "(no reason)",
        "|",
        formatBattleOutcomeForLog(state.lastBattleOutcome)
      );
      if (!ws.__pvDailySkipRestart) {
        maybeAutoRestartAfterClose();
      }
    };

    const connect = (wsUrl) => {
      const url = wsUrl || state.lastWsUrl || (state.token && state.runId ? buildWsUrl(state.token, state.runId) : null);
      if (!url) throw new Error("Arena wsUrl empty.");

      if (state.activeSocket && state.activeSocket.readyState <= 1) {
        try {
          state.activeSocket.__pvDailySkipRestart = true;
          state.activeSocket.close();
        } catch {}
      }

      const ws = new WebSocket(url);
      ws.addEventListener("open", () => {
        state.activeSocket = ws;
        state.lastBattleOutcome = null;
        log("WSS open:", url);
        sendInitialPositionOnce();
        runLoop();
      });
      ws.addEventListener("message", (ev) => parseIncoming(ev.data));
      ws.addEventListener("close", (ev) => handleSocketClose(ev, ws));
      ws.addEventListener("error", (ev) => warn("WSS error:", ev));
      return ws;
    };

    const startArena = async ({ token, autoConnect = true } = {}) => {
      if (state.startInFlight) {
        warn("startArena already in progress.");
        return state.lastStart;
      }

      const detected = detectToken({ activeWsUrl: state.activeSocket?.url, lastWsUrl: state.lastWsUrl });
      const finalToken = normalizeToken(token || state.token || detected.token);
      if (!finalToken) throw new Error("Token not found for arena.");
      if (!token && detected.source) {
        log(`Token detected from ${detected.source}`);
      }

      state.startInFlight = true;
      log(`Start request -> ${cfg.startPath}`);

      try {
        await ensureNoActiveArenaBeforeStart();
        if (cfg.autoUpgradeEfficiencyBeforeStart) {
          log("[PREP] Upgrade efficiency before weapon upgrade...");
          await upgradeMiningEfficiency(finalToken, "pre-weapon");
        } else {
          log("[PREP] Skip upgrade efficiency before arena (disabled by config).");
        }
        if (cfg.autoUpgradeWeaponBeforeStart) {
          log("[PREP] Upgrade bow attack before arena start...");
          await upgradeBowAttack(finalToken, "pre-start");
        } else {
          log("[PREP] Skip bow attack upgrade before arena (disabled by config).");
        }

        const controller = new AbortController();
        const timeoutId = setTimeout(() => controller.abort(), cfg.startTimeoutMs);
        let res;

        try {
          res = await fetch(cfg.startPath, {
            method: "POST",
            headers: {
              accept: "*/*",
              "content-type": "application/json",
              authorization: `Bearer ${finalToken}`
            },
            credentials: "include",
            body: "{}",
            signal: controller.signal
          });
        } catch (err) {
          const msg = err?.name === "AbortError"
            ? `Start timeout > ${cfg.startTimeoutMs}ms`
            : `Start fetch error: ${err?.message || String(err)}`;
          state.lastStart = {
            time: new Date().toISOString(),
            status: null,
            ok: false,
            error: msg
          };
          warn(msg);
          return state.lastStart;
        } finally {
          clearTimeout(timeoutId);
        }

        const text = await res.text();
        const data = safeJson(text);
        state.lastStart = {
          time: new Date().toISOString(),
          status: res.status,
          ok: res.ok,
          data
        };

        if (!res.ok || !data || typeof data !== "object" || !data.runId) {
          const isCooldown = data?.error === "cooldown" && typeof data?.cooldownUntil === "string";
          if (isCooldown) {
            const cooldownScheduled = scheduleCooldownRetry(finalToken, data.cooldownUntil);
            const result = {
              ...state.lastStart,
              cooldownScheduled,
              cooldownRetryAt: state.cooldownRetryAt
            };
            warn("Arena start cooldown:", result);
            return result;
          }
          warn("Arena start failed:", state.lastStart);
          return state.lastStart;
        }

        clearCooldownRetry();
        state.token = finalToken;
        state.runId = data.runId;
        state.battlesLeft = Number.isFinite(Number(data.battlesLeft)) ? Number(data.battlesLeft) : null;
        state.maxBattlesPerDay = Number.isFinite(Number(data.maxBattlesPerDay)) ? Number(data.maxBattlesPerDay) : null;
        state.lastWsUrl = buildWsUrl(finalToken, data.runId);

        log("Arena start OK. runId:", state.runId, "| battlesLeft:", state.battlesLeft);
        if (autoConnect) connect(state.lastWsUrl);

        return {
          ...state.lastStart,
          runId: state.runId,
          wsUrl: state.lastWsUrl
        };
      } finally {
        state.startInFlight = false;
      }
    };

    const stop = (reason = "manual_stop") => {
      state.stopped = true;
      clearCooldownRetry();
      if (state.activeSocket && state.activeSocket.readyState <= 1) {
        try {
          state.activeSocket.__pvDailySkipRestart = true;
          state.activeSocket.close();
        } catch {}
      }
      finishDaily(reason);
    };

    const runDaily = async () => {
      if (state.dailyInProgress) return state.dailyPromise;

      state.stopped = false;
      state.autoRestarting = false;
      state.autoRestartCount = 0;
      state.dailyInProgress = true;
      state.dailyPromise = new Promise((resolve) => {
        state.dailyResolve = resolve;
      });

      try {
        const first = await startArena({ autoConnect: true });
        if (!first?.ok) {
          if (first?.cooldownScheduled) {
            log("Initial start hit cooldown. Waiting retry at:", first.cooldownRetryAt);
          } else {
            const firstLeft = Number(first?.data?.battlesLeft);
            if (Number.isFinite(firstLeft) && firstLeft <= 0) {
              finishDaily("daily_limit_reached", first);
            } else {
              finishDaily("start_failed", first);
            }
          }
        }
      } catch (e) {
        finishDaily("start_exception", { message: e?.message || String(e) });
      }

      return state.dailyPromise;
    };

    const status = () => ({
      dailyInProgress: state.dailyInProgress,
      stopped: state.stopped,
      runningLoop: state.runningLoop,
      startInFlight: state.startInFlight,
      autoRestarting: state.autoRestarting,
      autoRestartCount: state.autoRestartCount,
      battlesLeft: state.battlesLeft,
      maxBattlesPerDay: state.maxBattlesPerDay,
      runId: state.runId,
      wsUrl: state.lastWsUrl,
      wsReadyState: state.activeSocket?.readyState ?? null,
      awaitingActiveArena: state.awaitingActiveArena,
      cooldownRetryAt: state.cooldownRetryAt,
      cooldownRetryScheduled: Boolean(state.cooldownRetryTimer),
      lastBattleOutcome: state.lastBattleOutcome
    });

    return {
      runDaily,
      stop,
      status
    };
  }

  function createMiningModule() {
    const cfg = CFG.mining;
    const log = (...args) => baseLog("[Mining]", ...args);
    const warn = (...args) => baseWarn("[Mining]", ...args);

    const state = {
      running: false,
      done: false,
      reason: null,
      promise: null,
      resolve: null
    };

    let rockIndex = null;
    let successCount = 0;
    let nextSellAt = cfg.sellEverySuccess;
    let lastNearFullSellAttemptAt = 0;
    let timedOneShotSellTriggered = false;

    const getToken = () => {
      const detected = detectToken();
      return normalizeToken(detected.token);
    };

    const getDelay = () => {
      if (cfg.mode === "fast") return randInt(1000, 1500);
      return randInt(3000, 5000);
    };

    const localDayKey = (date = new Date()) =>
      `${date.getFullYear()}-${String(date.getMonth() + 1).padStart(2, "0")}-${String(date.getDate()).padStart(2, "0")}`;

    const sellStartLabel = () =>
      `${String(cfg.autoSellStartTime.hour).padStart(2, "0")}:${String(cfg.autoSellStartTime.minute).padStart(2, "0")}`;

    const isAutoSellTimeReached = (at = new Date()) => {
      if (!cfg.autoSellEnabled) return false;
      if (!cfg.autoSellAfterTimeEnabled) return true;
      const gateAt = new Date(at);
      gateAt.setHours(cfg.autoSellStartTime.hour, cfg.autoSellStartTime.minute, 0, 0);
      return at.getTime() >= gateAt.getTime();
    };

    const stop = (reason) => {
      if (state.done) return;
      state.running = false;
      state.done = true;
      state.reason = reason || "stopped";
      log("Stop:", state.reason);
      if (typeof state.resolve === "function") {
        state.resolve({ ok: true, reason: state.reason });
      }
      state.resolve = null;
      state.promise = null;
    };

    const getCapacity = (snapshot) => Number(snapshot?.character?.capacity ?? snapshot?.capacity ?? 0);
    const getLevel = (snapshot) => Number(snapshot?.character?.level ?? snapshot?.level ?? 0);
    const getDailyMined = (snapshot) => Number(snapshot?.character?.dailyMinedToday ?? snapshot?.dailyMinedToday ?? 0);
    const getDailyLimit = (snapshot) => Number(snapshot?.character?.dailyMiningLimit ?? snapshot?.dailyMiningLimit ?? 0);
    const countFilledSlots = (slots = []) => slots.filter((s) => s.resourceType !== "empty").length;
    const getBagUsed = (snapshot) => {
      const slots = snapshot?.slots || [];
      return slots.reduce((sum, s) => {
        if (!s || s.resourceType === "empty") return sum;
        const count = Number(s.count || 0);
        return sum + (Number.isFinite(count) && count > 0 ? count : 0);
      }, 0);
    };

    const sumResources = (slots = []) => {
      const totals = { gold: 0, silver: 0, bronze: 0 };
      for (const s of slots) {
        if (Object.prototype.hasOwnProperty.call(totals, s.resourceType)) {
          totals[s.resourceType] += Number(s.count || 0);
        }
      }
      return totals;
    };

    const salePointsInfo = (sellData) => {
      const yourPoints = Number(sellData?.window?.yourPoints ?? sellData?.sale?.points);
      const totalPoints = Number(sellData?.window?.totalPoints);
      if (!Number.isFinite(yourPoints)) return "";
      return ` | Points ${yourPoints}/${Number.isFinite(totalPoints) ? totalPoints : "?"}`;
    };

    const fetchCharacter = async (token) => {
      try {
        const res = await fetch("/api/character", {
          credentials: "include",
          headers: { Authorization: `Bearer ${token}` }
        });
        if (!res.ok) return null;
        return await res.json();
      } catch {
        return null;
      }
    };

    const logCharacter = (data) => {
      if (!data) return;
      const totals = sumResources(data.slots || []);
      const capacity = getCapacity(data);
      const used = getBagUsed(data);
      const filledSlots = countFilledSlots(data.slots || []);
      log(
        `GOLD:${totals.gold} | SILVER:${totals.silver} | BRONZE:${totals.bronze} | Bag:${used}/${capacity} (slots:${filledSlots}) | Daily:${getDailyMined(data)}/${getDailyLimit(data)}`
      );
    };

    const upgradeMiningEfficiency = async (token, reason = "unspecified") => {
      try {
        const res = await fetch("/api/character/upgrade-mining-efficiency", {
          method: "POST",
          credentials: "include",
          headers: {
            accept: "*/*",
            "Content-Type": "application/json",
            Authorization: `Bearer ${token}`
          },
          body: "{}"
        });

        const text = await res.text();
        if (!res.ok) {
          warn(`[UPGRADE] Failed (${res.status}) [${reason}]: ${text}`);
          return { ok: false, status: res.status, body: text };
        }

        log(`[UPGRADE] Success [${reason}]: ${text}`);
        return { ok: true, status: res.status, body: text };
      } catch (err) {
        warn(`[UPGRADE] Network error [${reason}]:`, err);
        return { ok: false, status: null, body: String(err) };
      }
    };

    const sellSilverBronzeAll = async (token, slots) => {
      const charData = await fetchCharacter(token);
      const source = charData || slots || {};
      const level = getLevel(source);
      if (level < 3) {
        log(`[SELL] Skip: level ${level} (unlock level 3).`);
        return { ok: true, sold: false, skippedByLevel: true };
      }

      let totals = sumResources(charData?.slots || []);
      if (!totals.silver && !totals.bronze) {
        totals = sumResources(source?.slots || []);
      }
      if (!totals.silver && !totals.bronze) {
        log("[SELL] No silver/bronze to sell.");
        return { ok: true, sold: false };
      }

      const payload = { gold: 0, silver: totals.silver, bronze: totals.bronze };
      try {
        const res = await fetch("/api/market/sell", {
          method: "POST",
          credentials: "include",
          headers: {
            "Content-Type": "application/json",
            Authorization: `Bearer ${token}`
          },
          body: JSON.stringify(payload)
        });
        const text = await res.text();
        const data = safeJson(text);
        if (!res.ok) {
          warn(`[SELL] Failed (${res.status}): ${text}`);
          return { ok: false, sold: false };
        }
        log(`[SELL] Sold silver=${payload.silver}, bronze=${payload.bronze}.${salePointsInfo(data)}`);
        if (cfg.autoUpgradeEfficiencyAfterSell) {
          await upgradeMiningEfficiency(token, "post-sell");
        } else {
          log("[SELL] Skip post-sell efficiency upgrade (disabled by config).");
        }
        return { ok: true, sold: true, status: res.status, body: text, data };
      } catch (err) {
        warn("[SELL] Network error:", err);
        return { ok: false, sold: false };
      }
    };

    const sellAllInventory = async (reason = "manual_sell_all") => {
      const token = getToken();
      if (!token) {
        warn(`[SELL-ALL] Token missing [${reason}]`);
        return { ok: false, sold: false, error: "token_missing" };
      }

      const charData = await fetchCharacter(token);
      if (!charData) {
        warn(`[SELL-ALL] Failed to fetch character [${reason}]`);
        return { ok: false, sold: false, error: "character_fetch_failed" };
      }

      const totals = sumResources(charData?.slots || []);
      if (!totals.gold && !totals.silver && !totals.bronze) {
        log(`[SELL-ALL] Nothing to sell [${reason}].`);
        return { ok: true, sold: false, totals };
      }

      const payload = {
        gold: totals.gold,
        silver: totals.silver,
        bronze: totals.bronze
      };

      try {
        const res = await fetch("/api/market/sell", {
          method: "POST",
          credentials: "include",
          headers: {
            "Content-Type": "application/json",
            Authorization: `Bearer ${token}`
          },
          body: JSON.stringify(payload)
        });
        const text = await res.text();
        const data = safeJson(text);
        if (!res.ok) {
          warn(`[SELL-ALL] Failed (${res.status}) [${reason}]: ${text}`);
          return { ok: false, sold: false, status: res.status, body: text, data, payload };
        }

        log(
          `[SELL-ALL] Sold gold=${payload.gold}, silver=${payload.silver}, bronze=${payload.bronze} ` +
          `[${reason}].${salePointsInfo(data)}`
        );
        return { ok: true, sold: true, status: res.status, body: text, data, payload };
      } catch (err) {
        warn(`[SELL-ALL] Network error [${reason}]:`, err);
        return { ok: false, sold: false, status: null, body: String(err), payload };
      }
    };

    const maybeTriggerTimedOneShotSell = async (token, reason = "unspecified") => {
      if (!cfg.autoSellEnabled) return false;
      if (!cfg.autoSellAfterTimeEnabled) return false;
      if (timedOneShotSellTriggered) return false;
      if (!isAutoSellTimeReached()) return false;

      timedOneShotSellTriggered = true;
      log(`[SELL-TIME] Time gate reached (${sellStartLabel()}). Trigger one-time sell [${reason}]...`);
      await sellSilverBronzeAll(token, null);
      return true;
    };

    const maybeAutoSell = async (token, snapshot) => {
      if (!cfg.autoSellEnabled) return;
      if (!isAutoSellTimeReached()) return;
      const capacity = getCapacity(snapshot);
      const used = getBagUsed(snapshot);
      const freeCapacity = Math.max(capacity - used, 0);

      if (capacity > 50) {
        if (successCount < nextSellAt) return;
        log(`[SELL] Trigger by successCount=${successCount}.`);
        const result = await sellSilverBronzeAll(token, snapshot);
        if (result.sold) {
          nextSellAt += cfg.sellEverySuccess;
        } else if (result.skippedByLevel) {
          nextSellAt = successCount + 5;
        }
        return;
      }

      if (freeCapacity > cfg.lowCapacityFreeSlotTrigger) return;
      if (successCount - lastNearFullSellAttemptAt < 3) return;

      lastNearFullSellAttemptAt = successCount;
      log(`[SELL] Trigger near-full (capacity ${capacity}, used ${used}, free ${freeCapacity}).`);
      await sellSilverBronzeAll(token, snapshot);
    };

    const extractSnapshotFromMining = (data) => ({
      character: data?.character || {},
      slots: data?.slots || [],
      dailyMinedToday: data?.dailyMinedToday,
      dailyMiningLimit: data?.dailyMiningLimit,
      capacity: data?.capacity
    });

    const isBackpackFull = (snapshot) => {
      const capacity = getCapacity(snapshot);
      if (!capacity) return false;
      return getBagUsed(snapshot) >= capacity;
    };

    const isDailyLimitReached = (snapshot) => {
      const limit = getDailyLimit(snapshot);
      if (!limit) return false;
      return getDailyMined(snapshot) >= limit;
    };

    const getLatestSnapshot = async (token, fallbackSnapshot) => {
      const latest = await fetchCharacter(token);
      return latest || fallbackSnapshot;
    };

    const fetchRocksSnapshot = async (token) => {
      try {
        const url = `${cfg.rocksSnapshotPath}?mapId=${encodeURIComponent(cfg.mapId)}`;
        const res = await fetch(url, {
          credentials: "include",
          headers: {
            accept: "*/*",
            Authorization: `Bearer ${token}`
          }
        });
        const text = await res.text();
        if (!res.ok) {
          warn(`[ROCK] Snapshot failed (${res.status}): ${text}`);
          return null;
        }
        const data = safeJson(text);
        if (!data || typeof data !== "object" || !Array.isArray(data.rocks)) {
          warn("[ROCK] Snapshot invalid payload:", data);
          return null;
        }
        return data.rocks;
      } catch (err) {
        warn("[ROCK] Snapshot network error:", err);
        return null;
      }
    };

    const pickRandomAliveRock = async (token, reason = "unspecified") => {
      const rocks = await fetchRocksSnapshot(token);
      if (!rocks) return { ok: false, waitMs: 5_000 };

      const available = rocks
        .filter((r) => {
          const idx = Number(r?.rockIndex);
          const hp = Number(r?.healthPct);
          return Number.isInteger(idx)
            && idx >= cfg.rockMin
            && idx <= cfg.rockMax
            && Number.isFinite(hp)
            && hp > 0;
        })
        .map((r) => ({ rockIndex: Number(r.rockIndex), healthPct: Number(r.healthPct) }));

      if (!available.length) {
        rockIndex = null;
        log(`[ROCK] All rocks healthPct=0. Wait ${Math.ceil(cfg.allRocksEmptyWaitMs / 60000)}m before retry.`);
        return { ok: false, waitMs: cfg.allRocksEmptyWaitMs, allDepleted: true };
      }

      const picked = available[randInt(0, available.length - 1)];
      rockIndex = picked.rockIndex;
      log(`[ROCK] Selected rock ${rockIndex} from snapshot [${reason}] (healthPct=${picked.healthPct}).`);
      return { ok: true, rockIndex: picked.rockIndex, healthPct: picked.healthPct };
    };

    const evaluateLimitsOrStop = async (token, fallbackSnapshot) => {
      const latest = await getLatestSnapshot(token, fallbackSnapshot);
      if (!latest) return;
      if (isDailyLimitReached(latest)) {
        stop("daily_limit_reached");
        return;
      }
      if (isBackpackFull(latest)) {
        const level = getLevel(latest);
        if (level < 3) {
          stop("backpack_full_level_lt_3_run_arena");
          return;
        }
        if (!cfg.autoSellEnabled) {
          stop("backpack_full_auto_sell_disabled");
          return;
        }
        if (!isAutoSellTimeReached()) {
          stop("backpack_full_auto_sell_time_not_reached");
          return;
        }

        log("[BAG] Full and level >= 3. Try sell silver/bronze first...");
        const sellResult = await sellSilverBronzeAll(token, latest);
        if (!sellResult?.sold) {
          stop("backpack_full_sell_failed");
          return;
        }

        const afterSell = await fetchCharacter(token);
        if (afterSell && isBackpackFull(afterSell)) {
          stop("backpack_still_full_after_sell");
          return;
        }

        if (afterSell) {
          logCharacter(afterSell);
          return;
        }
      }
      logCharacter(latest);
    };

    const hitLoop = async () => {
      if (!state.running) return;

      const token = getToken();
      if (!token) {
        stop("token_missing");
        return;
      }

      await maybeTriggerTimedOneShotSell(token, "hit_loop");

      if (!Number.isInteger(rockIndex) || rockIndex < cfg.rockMin || rockIndex > cfg.rockMax) {
        const selected = await pickRandomAliveRock(token, "no_current_target");
        if (!selected.ok) {
          if (!state.running) return;
          setTimeout(hitLoop, selected.waitMs);
          return;
        }
      }

      try {
        const res = await fetch("/api/mining/hit", {
          method: "POST",
          credentials: "include",
          headers: {
            "Content-Type": "application/json",
            Authorization: `Bearer ${token}`
          },
          body: JSON.stringify({
            mapId: cfg.mapId,
            rockIndex
          })
        });

        if (!res.ok) {
          const raw = await res.text();
          let errBody = null;
          try { errBody = JSON.parse(raw); } catch {}

          if (errBody?.error === "daily_limit_reached") {
            stop("daily_limit_reached");
            return;
          }
          if (errBody?.error === "inventory_full") {
            log("[BAG] /api/mining/hit returned inventory_full. Checking bag state...");
            await evaluateLimitsOrStop(token);
            if (!state.running) return;
            setTimeout(hitLoop, 1_000);
            return;
          }
          if (errBody?.error === "rock_depleted") {
            log(`[ROCK] Rock ${rockIndex} already depleted (non-200 response). Pick another rock.`);
            rockIndex = null;
            setTimeout(hitLoop, 1_000);
            return;
          }

          log(`[HIT] Rock ${rockIndex} unavailable (${res.status}) ${raw ? `| ${raw}` : ""}`);
          rockIndex = null;

          setTimeout(hitLoop, 1_000);
          return;
        }

        const data = await res.json();
        if (data?.error === "daily_limit_reached") {
          stop("daily_limit_reached");
          return;
        }
        if (data?.error === "inventory_full") {
          log("[BAG] Mining response returned inventory_full. Checking bag state...");
          await evaluateLimitsOrStop(token);
          if (!state.running) return;
          setTimeout(hitLoop, 1_000);
          return;
        }
        if (data?.error === "rock_depleted") {
          log(`[ROCK] Rock ${rockIndex} depleted. Pick another rock.`);
          rockIndex = null;
          setTimeout(hitLoop, 1_000);
          return;
        }

        successCount += 1;
        log(`[HIT] Success #${successCount} | Rock ${rockIndex} HP:${data.rockHealthPct}`);

        if (data.drop && data.drop.resourceType !== "nothing") {
          log(`[DROP] +${data.drop.count} ${data.drop.resourceType}`);
        }

        const snapshot = extractSnapshotFromMining(data);
        await maybeAutoSell(token, snapshot);
        await evaluateLimitsOrStop(token, snapshot);
        if (!state.running) return;

        if (data.rockHealthPct <= 0) {
          log(`[ROCK] Rock ${rockIndex} destroyed`);
          rockIndex = null;
        }
      } catch (err) {
        warn("[HIT] Network error:", err);
      }

      if (!state.running) return;
      setTimeout(hitLoop, getDelay());
    };

    const runDaily = async () => {
      if (state.running) return state.promise;

      state.running = true;
      state.done = false;
      state.reason = null;
      state.promise = new Promise((resolve) => {
        state.resolve = resolve;
      });

      rockIndex = null;
      successCount = 0;
      nextSellAt = cfg.sellEverySuccess;
      lastNearFullSellAttemptAt = 0;
      timedOneShotSellTriggered = false;

      const token = getToken();
      if (!token) {
        stop("token_missing");
        return state.promise;
      }

      if (cfg.autoUpgradeEfficiencyBeforeMining) {
        log("[START] Upgrade mining efficiency before mining...");
        await upgradeMiningEfficiency(token, "pre-mining");
      } else {
        log("[START] Skip pre-mining efficiency upgrade (disabled by config).");
      }

      await maybeTriggerTimedOneShotSell(token, "run_start");

      if (cfg.autoPreSellBeforeMining && cfg.autoSellEnabled && isAutoSellTimeReached()) {
        log("[START] Pre-sell silver+bronze before mining...");
        await sellSilverBronzeAll(token, null);
      } else {
        if (!cfg.autoPreSellBeforeMining || !cfg.autoSellEnabled) {
          log("[START] Skip pre-sell before mining (disabled by config).");
        } else {
          log(`[START] Skip pre-sell before mining (waiting sell time gate ${sellStartLabel()}).`);
        }
      }

      const charData = await fetchCharacter(token);
      if (charData) logCharacter(charData);

      log("[START] Mining loop started.");
      hitLoop();

      return state.promise;
    };

    const status = () => ({
      running: state.running,
      done: state.done,
      reason: state.reason
    });

    return {
      runDaily,
      stop,
      status,
      sellAllInventory
    };
  }

  const arena = createArenaModule();
  const mining = createMiningModule();

  const daily = {
    running: false,
    stopped: false,
    phase: "idle",
    promise: null,
    result: null,
    autoRerunTimer: null,
    autoRerunAt: null,
    autoRerunLastDayKey: null
  };

  const dayKey = (date) =>
    `${date.getFullYear()}-${String(date.getMonth() + 1).padStart(2, "0")}-${String(date.getDate()).padStart(2, "0")}`;

  const toLocalDateTime = (base, hour, minute, second = 0, ms = 0) => {
    const d = new Date(base);
    d.setHours(hour, minute, second, ms);
    return d;
  };

  const randomMsBetween = (fromMs, toMs) => {
    if (toMs <= fromMs) return fromMs;
    return Math.floor(Math.random() * (toMs - fromMs + 1)) + fromMs;
  };

  const clearAutoRerunTimer = () => {
    if (daily.autoRerunTimer) {
      clearTimeout(daily.autoRerunTimer);
      daily.autoRerunTimer = null;
    }
    daily.autoRerunAt = null;
  };

  const computeNextAutoRerunAt = (now = new Date()) => {
    const startCfg = CFG.autoRerunWindowStart;
    const endCfg = CFG.autoRerunWindowEnd;
    const todayStart = toLocalDateTime(now, startCfg.hour, startCfg.minute, 0, 0);
    const todayEnd = toLocalDateTime(now, endCfg.hour, endCfg.minute, 59, 999);
    const today = dayKey(now);

    if (now < todayStart) {
      const pickMs = randomMsBetween(todayStart.getTime(), todayEnd.getTime());
      return new Date(pickMs);
    }

    if (now <= todayEnd && daily.autoRerunLastDayKey !== today) {
      const fromMs = Math.max(now.getTime() + 1_000, todayStart.getTime());
      const pickMs = randomMsBetween(fromMs, todayEnd.getTime());
      return new Date(pickMs);
    }

    const tomorrow = new Date(now);
    tomorrow.setDate(tomorrow.getDate() + 1);
    const tomorrowStart = toLocalDateTime(tomorrow, startCfg.hour, startCfg.minute, 0, 0);
    const tomorrowEnd = toLocalDateTime(tomorrow, endCfg.hour, endCfg.minute, 59, 999);
    const pickMs = randomMsBetween(tomorrowStart.getTime(), tomorrowEnd.getTime());
    return new Date(pickMs);
  };

  const scheduleNextAutoRerun = (reason = "unspecified") => {
    if (!CFG.autoRerunWindowEnabled) return;
    clearAutoRerunTimer();

    const nowDate = new Date();
    const nextAt = computeNextAutoRerunAt(nowDate);
    const delayMs = Math.max(1_000, nextAt.getTime() - nowDate.getTime());

    daily.autoRerunAt = nextAt.toISOString();
    baseLog(`[AutoRerun] Next run scheduled (${reason}) at ${nextAt.toLocaleString()} (in ${Math.ceil(delayMs / 1000)}s).`);

    daily.autoRerunTimer = setTimeout(async () => {
      daily.autoRerunTimer = null;
      daily.autoRerunAt = null;
      const triggerDay = dayKey(new Date());
      if (daily.autoRerunLastDayKey === triggerDay) {
        scheduleNextAutoRerun("already_triggered_today");
        return;
      }
      daily.autoRerunLastDayKey = triggerDay;

      baseLog("[AutoRerun] Triggering scheduled daily run...");
      try {
        if (daily.running) {
          baseLog("[AutoRerun] Existing run detected, restarting flow.");
          stop("scheduled_restart");
          await sleep(2_000);
        }
        run().catch((e) => baseWarn("[AutoRerun] run() error:", e?.message || e));
      } finally {
        scheduleNextAutoRerun("post_trigger");
      }
    }, delayMs);
  };

  const stop = (reason = "manual_stop") => {
    daily.stopped = true;
    arena.stop(reason);
    mining.stop(reason);
    baseLog("Stop requested:", reason);
  };

  const run = async () => {
    if (daily.running) return daily.promise;

    daily.running = true;
    daily.stopped = false;
    daily.phase = "arena";
    daily.result = null;

    daily.promise = (async () => {
      let cooldownMiningStopTimer = null;
      let handledCooldownRetryAt = null;
      const clearCooldownMiningStopTimer = () => {
        if (cooldownMiningStopTimer) {
          clearTimeout(cooldownMiningStopTimer);
          cooldownMiningStopTimer = null;
        }
      };

      const maybeRunMiningDuringArenaCooldown = () => {
        const arenaStatus = arena.status();
        const retryAt = arenaStatus?.cooldownRetryAt;
        const scheduled = arenaStatus?.cooldownRetryScheduled;
        if (!scheduled || !retryAt) return;
        if (handledCooldownRetryAt === retryAt) return;

        const retryAtMs = Date.parse(retryAt);
        if (!Number.isFinite(retryAtMs)) return;
        const remainingMs = Math.max(0, retryAtMs - Date.now());
        handledCooldownRetryAt = retryAt;

        if (remainingMs < 3_000) return;

        baseLog(
          `[Arena->Mining] Arena cooldown detected (${Math.ceil(remainingMs / 1000)}s). ` +
          "Run mining sementara sampai cooldown selesai."
        );

        if (!mining.status().running) {
          mining.runDaily().catch((e) => baseWarn("[Arena->Mining] Mining temp run error:", e?.message || e));
        }

        clearCooldownMiningStopTimer();
        cooldownMiningStopTimer = setTimeout(() => {
          if (mining.status().running) {
            baseLog("[Arena->Mining] Cooldown arena selesai. Stop mining sementara untuk retry arena.");
            mining.stop("pause_for_arena_retry");
          }
        }, Math.max(1_000, remainingMs - 1_000));
      };

      try {
        baseLog("Daily pipeline start: arena -> mining");
        const arenaPromise = arena.runDaily();
        let arenaDone = false;
        let arenaResult = null;
        arenaPromise.then((res) => {
          arenaDone = true;
          arenaResult = res;
        });

        while (!arenaDone && !daily.stopped) {
          maybeRunMiningDuringArenaCooldown();
          await sleep(1_000);
        }

        if (!arenaDone) {
          arenaResult = await arenaPromise;
        }

        clearCooldownMiningStopTimer();
        if (mining.status().running) {
          baseLog("[Arena->Mining] Arena phase selesai. Stop mining sementara.");
          mining.stop("arena_finished");
        }

        if (daily.stopped) {
          return { ok: false, stopped: true, phase: "arena", arena: arenaResult, mining: null };
        }

        baseLog("Arena done:", arenaResult?.reason || "unknown");
        daily.phase = "wait_before_mining";
        baseLog(`Waiting ${CFG.betweenArenaAndMiningMs / 1000}s before mining...`);
        await sleep(CFG.betweenArenaAndMiningMs);

        if (daily.stopped) {
          return { ok: false, stopped: true, phase: "wait_before_mining", arena: arenaResult, mining: null };
        }

        daily.phase = "mining";
        const miningStatus = mining.status();
        let miningResult;
        if (miningStatus.reason === "daily_limit_reached" || miningStatus.reason === "backpack_full") {
          baseLog("Mining sudah selesai saat fase cooldown arena, skip mining final.");
          miningResult = { ok: true, reason: "already_completed_during_arena_cooldown", detail: miningStatus };
        } else {
          miningResult = await mining.runDaily();
        }

        daily.phase = "completed";
        daily.result = { ok: true, arena: arenaResult, mining: miningResult };
        baseLog("DAILY COMPLETE: arena + mining finished.");
        return daily.result;
      } catch (err) {
        daily.phase = "error";
        const result = { ok: false, error: err?.message || String(err) };
        daily.result = result;
        baseWarn("Daily pipeline error:", result.error);
        return result;
      } finally {
        clearCooldownMiningStopTimer();
        daily.running = false;
        daily.promise = null;
      }
    })();

    return daily.promise;
  };

  const status = () => ({
    running: daily.running,
    stopped: daily.stopped,
    phase: daily.phase,
    result: daily.result,
    autoRerunAt: daily.autoRerunAt,
    autoRerunEnabled: CFG.autoRerunWindowEnabled,
    autoRerunWindow: `${String(CFG.autoRerunWindowStart.hour).padStart(2, "0")}:${String(CFG.autoRerunWindowStart.minute).padStart(2, "0")} - ${String(CFG.autoRerunWindowEnd.hour).padStart(2, "0")}:${String(CFG.autoRerunWindowEnd.minute).padStart(2, "0")}`,
    arena: arena.status(),
    mining: mining.status()
  });

  const sellAll = async () => {
    const result = await mining.sellAllInventory("console_command");
    return result;
  };

  window.pvDaily = {
    run,
    stop,
    status,
    sellAll,
    sell_all: sellAll,
    config: CFG,
    version: BUILD_META.version,
    lastUpdated: BUILD_META.lastUpdated,
    meta: { ...BUILD_META }
  };
  window.sell_all = sellAll;

  baseLog("Ready.");
  baseLog(`Build: ${BUILD_META.version} | Last update: ${BUILD_META.lastUpdated}`);
  baseLog(`Flow: Arena daily -> wait ${CFG.betweenArenaAndMiningMs / 1000}s -> Mining daily -> complete log.`);
  baseLog("Commands: pvDaily.run(), pvDaily.stop(), pvDaily.status(), pvDaily.sell_all(), sell_all(), pvDaily.config");

  if (CFG.autoRerunWindowEnabled) {
    scheduleNextAutoRerun("init");
  }

  if (CFG.autoRunOnLoad) {
    run();
  }
})();
